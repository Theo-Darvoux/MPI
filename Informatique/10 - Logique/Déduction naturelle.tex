Le but de ce chapitre est de commencer à formaliser ce qu'est une preuve. On pourrait ainsi essayer d'en générer, ou d'utiliser ce formalisme pour vérifier des preuves. Une preuve au sens usuel est un raisonnement mathématique qui, à partir d'un ensemble d'hypothèses amène à une conclusion. 

\section{Rappels de logique propositionnelle.}

\begin{rappel}{Définition syntaxique.}{}
    Si $\V$ est un ensemble de variables propositionnelles, on peut définir des formules propositionnelles sur les variables de $\V$ par induction, pour $\phi$ et $\psi$ deux formules propositionnelles :\\
    $\hra$ $\top$ et $\bot$ sont des formules propositionnelles.\\
    $\hra$ $x\in\V$ est une formule propositionnelle.\\
    $\hra$ $\phi \land \psi$ et $\phi \lor \psi$ sont des formules propositionnelles.\\
    $\hra$ $\lnot\phi$ est une formule propositionnelle.\\
    On a défini les formules propositionnelles syntaxiquement : on peut simplement reconnaître les formules propositionnelles.
\end{rappel}

\vspace*{-0.9cm}

\begin{rappel}{Définition sémantique.}{}
    On associe aux formules propositionnelles une définition sémantique : on leur donne un sens.\\
    On définit les valuations : application de $\V$ dans $\{0,1\}$, et on l'étend aux formules propositionnelles inductivement:\\
    $\hra$ Si $\phi=\top$, on pose $v(\phi)=1$ et si $\phi=\bot$, $v(\phi)=0$.\\
    $\hra$ Si $\phi=x\in\V$, on pose $v(\phi)=v(x)$.\\
    $\hra$ Si $\phi=\phi_1\land\phi_2$, on pose $v(\phi)= v(\phi_1)v(\phi_2)$.\\
    $\hra$ Si $\phi=\phi_1\lor\phi_2$, on pose $v(\phi)=v(\phi_1)+v(\phi_2) - v(\phi_1)v(\phi_2)$.\\
    $\hra$ Si $\phi=\lnot\phi_1$, on pose $v(\phi) = 1 - v(\phi_1)$.
\end{rappel}

\vspace*{-0.9cm}

\begin{defi}{Modèle.}{}
    On note $\M(\phi)=\{v:\V\to\{0,1\}, ~ v(\phi)=1\}$, appelé \bf{modèle} de $\phi$. On peut le construire inductivement :\\
    $\hra$ $\M(\bot)=\0$ et $\M(\top)=\{v:\V\to\{0,1\}\}$.\\
    $\hra$ $\M(x)=\{v:\V\to\{0,1\}, ~ v(x)=1\}$.\\
    $\hra$ $\M(\phi_1\land\phi_2)=\M(\phi_1)\cap\M(\phi_2)$.\\
    $\hra$ $\M(\phi_1\lor\phi_2)=\M(\phi_1)\cup\M(\phi_2)$.\\
    $\hra$ $\M(\lnot\phi_1)=\ov{\M(\phi_1)}$.\\
    Si $\M(\phi)\neq\0$, on dit que $\phi$ est \bf{satisfiable}; si $\M(\phi)=\{v:\V\to\{0,1\}\}$, $\phi$ est une \bf{tautologie}.\\
    On dit que $\phi$ et $\psi$ sont \bf{sémantiquement équivalentes} si $\M(\phi)=\M(\psi)$, on note $\phi\equiv\psi$.
\end{defi}

\begin{ex}{}{}
    Puisque $\forall \phi,\psi ~ : ~ \left[(\phi \ra \psi) \equiv (\lnot \phi \lor \psi)\right]$, on peut montrer par induction que pour toute formule $\phi$, il existe une formule $\phi'$ telle que $\phi'\equiv \phi$ et n'utilisant pas le symbole $\ra$.
\end{ex}

\vspace*{-0.8cm}

\begin{ex}{}{}
    Puisque $\lnot(\lnot \phi \land \lnot \psi) \equiv (\phi \lor \psi)$, on peut démontrer que seuls $\land$ et $\lnot$ sont suffisants pour décrire toutes les formules propositionnelles (ou $\lor$ et $\lnot$).
\end{ex}

Lorsqu'on prouve quelque chose, on cherche à prouver que la formule sous-jacente est une tautologie.\\

\section{Déduction naturelle.}

\begin{defi}{}{}
    On appelle séquent un couple $(\G,\phi)$ où $\G$ est un ensemble de formules et $\phi$ une formule.\\
    On le note $\G\vdash\phi$, et si $\G=\0$, on notera $\vdash\phi$.
\end{defi}

\vspace*{-0.8cm}

\begin{defi}{}{}
    On appelle règle d'inférence un $(n+1)$-uplet de séquents et on la note : 
    \begin{center}
        \begin{prooftree}
            \AxiomC{$\G_1\vdash\phi_1$}
            \AxiomC{$\G_2\vdash\phi_2$}
            \AxiomC{$\cdots$}
            \AxiomC{$\G_n\vdash\phi_n$}
            \QuaternaryInfC{$\G\vdash\phi$}
        \end{prooftree}
    \end{center}
    Les $n$ premiers séquents sont les prémisses, le dernier est la conclusion.
\end{defi}

\vspace*{-0.8cm}

\begin{defi}{Élimination de l'implication.}{}
    On appelle élimination de l'implication la règle d'inférence :
    \begin{prooftree}
        \AxiomC{$\G\vdash\phi$}
        \AxiomC{$\G\vdash\phi\ra\psi$}
        \RightLabel{$\Rightarrow_e$}
        \BinaryInfC{$\G\vdash\psi$}
    \end{prooftree}
    Si avec les hypothèses $\G$ on prouve $\phi$, et que $\phi$ implique $\psi$, alors on peut prouver $\psi$ sous les hypothèses $\G$.
\end{defi}

\vspace*{-0.8cm}

\begin{defi}{Introduction de l'implication.}{}
    On appelle introduction de l'implication :
    \begin{prooftree}
        \AxiomC{$\G, \phi\vdash\psi$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\G\vdash\phi\ra\psi$}
    \end{prooftree}
    Si on arrive à prouver $\psi$ en supposant $\G$ et $\phi$, on arrive à prouver $\phi\ra\psi$ sans supposer $\phi$, seulement $\G$.
\end{defi}

\vspace*{-0.8cm}

\begin{defi}{Arbre de preuve.}{}
    Si $\G\vdash\phi$ est un séquent, on appelle arbre de preuve de $\G\vdash\phi$ :\\
    $\bullet$ Si $\G\vdash\phi$ appartient au systèmé déductif, l'arbre est $\ov{\G\vdash\phi}$.\\
    $\bullet$ Sinon :
    \begin{prooftree}
        \AxiomC{$\G_1\vdash\phi_1$}
        \AxiomC{$\G_2\vdash\phi_2$}
        \AxiomC{$\cdots$}
        \AxiomC{$\G_n\vdash\phi_n$}
        \QuaternaryInfC{$\G\vdash\phi$}
    \end{prooftree}
    C'est une définition par récurrence.
\end{defi}

\begin{ex}{}{}
    Trouvons un arbre de preuve pour $\vdash(A\ra B)\ra(B\ra C)\ra(A\ra C)$.
    \tcblower
    \vspace*{-1cm}
    \begin{prooftree}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G\vdash B\ra C$}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G\vdash A \ra B$}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G\vdash A$}
        \RightLabel{$\Rightarrow$e}
        \BinaryInfC{$\G\vdash B$}
        \RightLabel{$\Rightarrow$e}
        \BinaryInfC{$\G:= ~ A\ra B, B\ra C, A\vdash C$}
        \RightLabel{$\Rightarrow$i}
        \UnaryInfC{$A\ra B, B\ra C \vdash A\ra C$}
        \RightLabel{$\Rightarrow$i}
        \UnaryInfC{$A\ra B \vdash (B\ra C)\ra(A\ra C)$}
        \RightLabel{$\Rightarrow$i}
        \UnaryInfC{$\vdash(A\ra B)\ra(B\ra C)\ra(A\ra C)$}
    \end{prooftree}
\end{ex}

\vspace*{-0.9cm}

\begin{defi}{Règles d'inférence.}{}
    On définit d'autres règles d'inférence pour les autres constructeurs logiques :
    \vspace*{0.2cm}
    \begin{center}
        \begin{tabular}{| c | c c c |}
            \hline&&&\\
            $\land$ &
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi$}
                \AxiomC{$\G\vdash\psi$}
                \RightLabel{$\land_{i}$}
                \BinaryInfC{$\G\vdash\phi\land\psi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi\land\psi$}
                \RightLabel{$\land_{eg}$}
                \UnaryInfC{$\G\vdash\phi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi\land\psi$}
                \RightLabel{$\land_{ed}$}
                \UnaryInfC{$\G\vdash\psi$}
            \end{bprooftree}\\&&&\\\hline
        \end{tabular}
    \end{center}
    \vspace*{0.5cm}
    \begin{center}
        \begin{tabular}{|c|c c c|}
            \hline &&&\\
            $\lor$ & 
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi$}
                \RightLabel{$\lor_{ig}$}
                \UnaryInfC{$\G\vdash\phi\lor\psi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi$}
                \RightLabel{$\lor_{id}$}
                \UnaryInfC{$\G\vdash\phi\lor\psi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash\phi\lor\psi$}
                \AxiomC{$\G,\phi\vdash \a$}
                \AxiomC{$\G,\psi\vdash \a$}
                \RightLabel{$\lor_e$}
                \TrinaryInfC{$\G\vdash\a$}
            \end{bprooftree}\\&&&\\
            \hline
        \end{tabular}
    \end{center}
    \vspace*{0.5cm}
    \begin{center}
        \begin{tabular}{|c|c c|}
            \hline &&\\
            $\lnot$ & 
            \begin{bprooftree}
                \AxiomC{$\G\vdash \lnot \phi$}
                \AxiomC{$\G\vdash \phi$}
                \RightLabel{$\lnot_e$}
                \BinaryInfC{$\G\vdash\bot$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G,\phi\vdash\bot$}
                \RightLabel{$\lnot_{i}$}
                \UnaryInfC{$\G\vdash\lnot \phi$}
            \end{bprooftree}\\&&\\
            \hline
        \end{tabular}
    \end{center}
    
    \vspace*{0.5cm}

    \begin{center}
        \begin{tabular}{c c c c}
            \begin{bprooftree}
                \AxiomC{}
                \RightLabel{$\top_i$}
                \UnaryInfC{$\G\vdash \top$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash \psi$}
                \RightLabel{aff}
                \UnaryInfC{$\G,\phi \vdash \psi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G,\lnot\phi\vdash\bot$}
                \RightLabel{abs}
                \UnaryInfC{$\G\vdash \psi$}
            \end{bprooftree}
            &
            \begin{bprooftree}
                \AxiomC{$\G\vdash \bot$}
                \RightLabel{$\bot_e$}
                \UnaryInfC{$\G\vdash\phi$}
            \end{bprooftree}
        \end{tabular}
    \end{center}
    La logique minimale (Lm) est le système composé de ces règles, sauf $\bot_e$ et abs.\\
    La logique intuitionniste (Li) est le système composé de ces règles, sauf abs.\\
    La logique classique (Lc) est le système composé de ces règles, sauf $\bot_e$.
\end{defi}

\vspace*{-0.9cm}

\begin{ex}{}{}
    Prouvons $\vdash(A\lor B) \ra (B\lor A)$.
    \tcblower
    \vspace*{-1cm}
    \begin{prooftree}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$A\lor B \vdash A\lor B$}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$A\lor B, A\vdash A$}
        \RightLabel{$\lor_{id}$}
        \UnaryInfC{$A\lor B, A\vdash B\lor A$}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$A\lor B, B\vdash B$}
        \RightLabel{$\lor_{ig}$}
        \UnaryInfC{$A\lor B, B\vdash B\lor A$}
        \TrinaryInfC{$A\lor B \vdash B \lor A$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash(A\lor B)\ra(B\lor A)$}
    \end{prooftree}
\end{ex}

\vspace*{-0.9cm}

\begin{ex}{}{}
    Prouvons $\vdash (A\lor B)\land C \ra (A\land C)\lor (B \land C)$.
    \tcblower
    \vspace*{-0.5cm}
    \tiny
    \begin{prooftree}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$(A\lor B) \land C \vdash (A \lor B) \land C$}
        \RightLabel{$\land_{eg}$}
        \UnaryInfC{$(A\lor B)\land C \vdash (A \lor B)$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$(A\lor B)\land C, A \vdash A$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$(A\lor B)\land C, C \vdash (A\lor B)\land C$}
        \RightLabel{$\land_{ed}$}
        \UnaryInfC{$(A\lor B)\land C,A \vdash C$}
        \RightLabel{$\land_{ed}$}
        \BinaryInfC{$(A\lor B)\land C, A \vdash A \land C$}
        \RightLabel{$\land_{ed}$}
        \UnaryInfC{$(A\lor B)\land C \vdash C, A\vdash (A\land C)\lor (B\land C)$}
        \AxiomC{$\vdots$}
        \UnaryInfC{$(A\lor B)\land C, B \vdash (A\land C)\lor (B\land C)$}
        \RightLabel{$\land_{ed}$}
        \TrinaryInfC{$(A\lor B)\land C \vdash (A\land C)\lor(B\land C)$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash(A\lor B)\land C \ra (A\land C)\lor(B\land C)$}
    \end{prooftree}
\end{ex}

\begin{ex}{}{}
    Montrons que $\lnot P \iff (P \ra \bot)$.
    \tcblower
    \vspace*{-1cm}
    \begin{prooftree}
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\lnot P, P \vdash \lnot P$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\lnot P, P \vdash P$}

        \RightLabel{$\Rightarrow_i$}
        \BinaryInfC{$\lnot P\vdash P\ra \bot$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash\lnot P \ra (P\ra \bot)$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$P\ra \bot,P\vdash P\ra \bot$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$P\ra \bot, P\vdash P$}

        \RightLabel{$\Rightarrow_e$}
        \BinaryInfC{$P\ra \bot, P \vdash \bot$}
        \RightLabel{$\lnot_i$}
        \UnaryInfC{$P\ra \bot \vdash \lnot P$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash(P\ra \bot) \ra \lnot P$}

        \RightLabel{$\land_i$}
        \BinaryInfC{$\vdash \lnot P \iff (P \ra \bot)$}
    \end{prooftree}
\end{ex}

\begin{ex}{}{}
    Montrons que $\lnot \top \iff \bot$.
    \tcblower
    \vspace*{-1cm}
    \begin{prooftree}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\lnot \top \vdash \lnot \top$}

        \AxiomC{}
        \RightLabel{$\top_i$}
        \UnaryInfC{$\lnot\top \vdash \top$}

        \RightLabel{$\land_e$}
        \BinaryInfC{$\lnot \top \vdash \bot$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash \lnot \top \ra \bot$}


        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\bot,\top\vdash\bot$}
        \RightLabel{$\lnot_i$}
        \UnaryInfC{$\bot \vdash \lnot \top$}
        \RightLabel{$\Rightarrow_i$}
        \UnaryInfC{$\vdash \bot \ra \lnot \top$}

        \RightLabel{$\land_i$}
        \BinaryInfC{$\vdash \lnot \top \iff \bot$}
    \end{prooftree}
\end{ex}

\vspace*{-0.8cm}

\begin{defi}{}{}
    Une règle est dite dérivable dans un système déductif si sa conclusion est prouvable dans ce système auquel on a ajouté ses prémisses sous forme d'axiomes. Si un séquent est prouvable à l'aide d'une règle dérivable, il est prouvable sans. L'utilisation de la règle dérivable ne change donc pas quels séquents sont prouvables. On peut donc les utiliser pour raccourcir des arbres de preuves. En effet, si une règle est dérivable et permet de prouver un séquent, celui-ci amet un arbre de preuve qui l'utilise. Chaque occurence de la règle dérivable peut être remplacée par l'arbre prouvant qu'elle est dérivable. On le formaliserait par induction.
\end{defi}

\vspace*{-0.8cm}

\begin{prop}{}{}
    La règle $\bot_e$ est dérivable en logique classique.
    \begin{prooftree}
        \AxiomC{$\G\vdash\bot$}
        \RightLabel{aff}
        \UnaryInfC{$\G,\lnot\phi\vdash\bot$}
        \RightLabel{abs}
        \UnaryInfC{$\G\vdash\phi$}
    \end{prooftree}
    Tout séquent de la logique intuitionniste est donc prouvable en logique classique.
\end{prop}

\vspace*{-0.8cm}

\begin{prop}{}{}
    Dans Lm, on peut prouver $\G\vdash P\ra \lnot\lnot P$, mais pas $\G\vdash\lnot\lnot P\ra P$.
\end{prop}

\vspace*{-0.8cm}

\begin{defi}{}{}
    Deux systèmes déductifs sont dits équivalents si touts les séquents prouvables dans l'un le sont aussi dans l'autre.
\end{defi}

\vspace*{-0.8cm}

\begin{meth}{}{}
    Pour prouver que deux systèmes $S_1$ et $S_2$ sont équivalents, on montre :\\
    $\hra$ Que toutes les règles de $S_1$ sont dérivables dans $S_2$.\\
    $\hra$ Que toutes les règles de $S_2$ sont dérivables dans $S_1$.
\end{meth}

\vspace*{-0.8cm}

\begin{defi}{}{}
    On note $\lnot\lnot_e$ la règle suivante :
    \begin{prooftree}
        \AxiomC{$\G\vdash\lnot\lnot \phi$}
        \RightLabel{$\lnot\lnot_e$}
        \UnaryInfC{$\G\vdash \phi$}
    \end{prooftree}
\end{defi}

\begin{prop}{}{}
    La logique intuitionniste munie de $\lnot\lnot_e$ est équivalente à la logique classique.
    \tcblower
    \vspace*{-.5cm}
    \begin{center}
        \begin{bprooftree}
            \AxiomC{}
            \RightLabel{ax}
            \UnaryInfC{$\G\vdash\lnot\lnot A$}
            \RightLabel{aff}
            \UnaryInfC{$\G,\lnot A \vdash \lnot \lnot A$}

            \AxiomC{}
            \RightLabel{ax}
            \UnaryInfC{$\G,\lnot A\vdash \lnot A$}

            \RightLabel{$\lnot_e$}
            \BinaryInfC{$\G,\lnot A \vdash \bot$}
            \RightLabel{abs}
            \UnaryInfC{$\G\vdash A$}
        \end{bprooftree}
        \quad\et\quad
        \begin{bprooftree}
            \AxiomC{}
            \RightLabel{ax}
            \UnaryInfC{$\G,\lnot P\vdash \bot$}
            \RightLabel{$\lnot_i$}
            \UnaryInfC{$\G\vdash \lnot\lnot P$}
            \RightLabel{$\lnot\lnot_e$}
            \UnaryInfC{$\G\vdash P$}
        \end{bprooftree}
    \end{center}
    Ainsi, $\lnot\lnot_e$ est dérivable en logique classique et abs est dérivable en logique intuitionniste munie de $\lnot\lnot_e$.\\
    Les deux systèmes sont équivalents puisque les règles $\lnot\lnot_e$, $\bot_e$ et abs sont les seules qui les distinguent.\\
    Pour dériver abs, on n'a pas utilisé $\bot_e$. La logique minimale munie de $\lnot\lnot_e$ est aussi équivalente à ces systèmes.
\end{prop}

\begin{exercice}{}{}
    On définit la règle suivante :
    \begin{prooftree}
        \AxiomC{$\G\vdash \lnot Q \ra \lnot P$}
        \RightLabel{cc}
        \UnaryInfC{$\G\vdash P\ra Q$}
    \end{prooftree}
    Montrer que Li muni de cc est équivalent à Lc. Est-ce aussi le cas de Lm ?
\end{exercice}

\begin{prop}{}{}
    En logique classique, on a bien la règle du tiers-exclu :
    \begin{prooftree}
        \AxiomC{}
        \RightLabel{$t_e$}
        \UnaryInfC{$\G\vdash A\lor \lnot A$}
    \end{prooftree}
    \tcblower
    \vspace*{-1cm}
    \begin{prooftree}
        
        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G,\lnot(A\lor\lnot A)\vdash\lnot(A\lor \lnot A)$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G'\vdash \lnot(A\lor \lnot A)$}

        \AxiomC{}
        \RightLabel{ax}
        \UnaryInfC{$\G'\vdash A$}
        \RightLabel{$\lor_{id}$}
        \UnaryInfC{$\G'\vdash\lnot A\lor\lnot A$}
        
        \RightLabel{$\lnot_e$}
        \BinaryInfC{$\G,\lnot(A\lor\lnot A),\lnot A\vdash\bot$}
        \RightLabel{$\lnot_i$}
        \UnaryInfC{$\G,\lnot(A\lor\lnot A)\vdash \lnot A$}
        \RightLabel{$\lor_{id}$}
        \UnaryInfC{$\G,\lnot(A\lor\lnot A)\vdash A\lor \lnot A$}

        \RightLabel{$\lnot_e$}
        \BinaryInfC{$\G,\lnot(A\lor \lnot A)\vdash \bot$}
        \RightLabel{abs}
        \UnaryInfC{$\G\vdash A\lor\lnot A$}
    \end{prooftree}
\end{prop}
