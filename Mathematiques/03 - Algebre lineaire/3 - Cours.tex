\section{Applications linûˋaires.}

\begin{defi}{}{}
    Soit $f\in\F(E,F)$. On dit que $f$ est \bf{linûˋaire} si et seulement si :
    \begin{equation*}
        \forall (x,y) \in E^2, ~ \forall \l \in \K \quad f(x+\l y)=f(x) + \l f(y).
    \end{equation*}
\end{defi}

\begin{prop}{}{}
    Soit $f\in\L(E,F)$. Alors $\Ker(f)$ et $\Im(f)$ sont des s.e.v.
    \tcblower
    \boxed{\Ker} $0_E\in\Ker(f)$. Soient $x,y\in \Ker(f)$ et $\l\in\K$. On a $f(x+\l y) = f(x)+\l f(y)=0$ donc $x+\l y \in \Ker(f)$.\\
    \boxed{\Im} $0_F\in\Im(f)$. Soient $x,y\in\Im(f)$ et $\l\in\K$. On a $\exists a,b\in E, ~ f(a)=x \et f(b)=y$. Donc $f(a+\l b)=x+\l y$.
\end{prop}

\begin{prop}{}{}
    Soient $f,g\in\L(E,F)$ de rangs finis $p$ et $q$. Alors $\rg(f+g)\leq\rg(f)+\rg(g)$.
    \tcblower
    On a bien $f+g$ une application linûˋaire de $E$ dans $F$, puis $\Im(f+g)\subset\Im(f)+\Im(g)$.\\
    D'aprû´s Grassmann, on a $\dim(\Im f + \Im g)=\dim(\Im f) + \dim(\Im g) - \dim(\Im f \cap \Im g)$.\\
    Enfin, $\rg(f+g)\leq\dim(\Im f + \Im g)=\rg(f) + \rg(g)  - \dim(\Im f \cap \Im g)\leq \rg(f) + \rg(g)$.
\end{prop}

\begin{thm}{du rang.}{}
    On suppose $E$ de dimension finie $n$. Soit $f\in\L(E,F)$, alors :
    \begin{equation*}
        \dim(E)=\dim(\Ker f)+\rg(f).
    \end{equation*}
    \tcblower
    On a $\Ker(f)$ un sev de $E$ donc de dimension finie $p\leq n$.\\
    Soit $(a_1,...,a_p)$ une base de $\Ker(f)$, on la complû´te en une base $(a_1,...,a_n)$ de $E$.\\
    On a alors $\Im(f)=\Vect(f(a_1),...,f(a_n))=\Vect(f(a_{p+1}),...,f(a_n))$ car $\forall i \in \lb1,p\rb, ~ f(a_i)=0$.\\
    Vûˋrifions que $\B'=(f(a_{p+1}),...,f(a_n))$ est une base de $\Im(f)$.\\
    --- Elle est gûˋnûˋratrice d'aprû´s ce qui prûˋcû´de.\\
    --- Soient $(\l_{p+1},...,\l_n)\in\K^{n-p}$ tels que $\l_{p+1}f(a_{p+1})+...+\l_nf(a_n)=0$. Alors $f(\l_{p+1}a_{p+1}+...+\l_na_n)=0$.\\
    i.e. $\l_{p+1}a_{p+1}+...+\l_na_n\in\Ker(f)$ donc $\exists(\l_1,...,\l_p)\in\K^p$ tels que $\l_{p+1}a_{p+1}+...+\l_na_n=\l_1a_1+...+\l_pa_p$.\\
    Donc puisque $(a_1,...,a_n)$ est libre, on a $\l_1=...=\l_n=0_\K$. Donc $\B'$ est libre.\n
    Alors $\B'$ est base de $\Im(f)$. Alors $\rg(f)=n-p$ et $\dim(\Ker f)=p$, on a bien $\dim(E)=\dim(\Ker f)+\rg(f)$.
\end{thm}

\begin{prop}{}{}
    Soit $f\in\L(E,F)$ puis $g\in\L(F,G)$.
    \begin{enumerate}
        \item Si $\rg(g)$ est fini, alors $\rg(g\circ f)$ est fini et $\rg(g\circ f)\leq\rg(g)$.
        \item Si $\rg(f)$ est fini, alors $\rg(g\circ f)$ est fini et $\rg(g\circ f)\leq\rg(f)$.
    \end{enumerate}
    On retiendra que $\rg(g\circ f)\leq\min(\rg(f),\rg(g))$.
    \tcblower
    On a bien $g\circ f$ une application linûˋaire de $E$ dans $G$. .\\
    \boxed{1.} Supposons $g$ de rang fini $n$. On a $\Im(g\circ f)\subset\Im(g)$, alors $\rg(g\circ f)\leq \rg(g)$.\\
    \boxed{2.} Supposons $f$ de rang fini $p$. On considû´re $g_{|\Im(f)}$ restriction de $g$, c'est toujours une application linûˋaire.\\
    Alors par thûˋorû´me du rang : $\dim(\Im f) = \dim(\Ker g_{|\Im f}) + \rg(g_{|\Im(f)})=\dim(\Ker g \cap \Im f)+\rg(g\circ f)\geq\rg(g\circ f)$.
\end{prop}

\begin{prop}{}{}
    Soit $f\in\L(E,F)$.
    \begin{enumerate}
        \item $f$ est injective $\iff$ $\Ker(f)\subset\{0_E\}$.
        \item $f$ est surjective $\iff$ $\Im(f)=F$ $\iff$ $F\subset\Im(f)$.
    \end{enumerate}
\end{prop}

\begin{thm}{}{}
    On suppose $\dim(E)=\dim(F)$ de dimensions finies. Soit $f\in\L(E,F)$.\\
    Alors $f$ injective $\iff$ $f$ surjective $\iff$ $f$ bijective.
    \tcblower
    D'aprû´s le thûˋorû´me du rang, $\dim(E)=\dim(\Ker f)+\rg(f)=\dim(F)$.\\
    On a $f$ injective $\iff \dim(\Ker f) = 0 \iff \rg(f)=\dim(F) \iff \Im(f)=F \iff f$ surjective.
\end{thm}

\begin{prop}{}{}
    Soit $f\in\L(E,F)$.
    \begin{enumerate}
        \item L'image par $f$ de toute famille liûˋe est liûˋe.
        \item Si $f$ est injective, alors l'image par $f$ de toute famille libre est libre.
        \item Si $f$ est surjective, alors l'image par $f$ de toute famille gûˋnûˋratrice est gûˋnûˋratrice.
        \item $f$ est bijective si et seulement si l'image par $f$ de toute base de $E$ est une base de $F$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Soit $(a_1,...,a_p)$ liûˋe de vecteurs de $E$ : $\exists(\l_1,...,\l_p)\in\K^p$ non tous nuls tels que $\sum_{i=1}^p\l_ia_i=0$.\\
    On a $f(\sum_{i=1}^p\l_ia_i)=\sum_{i=1}^p\l_if(a_i)=0$ donc $(f(a_1),...,f(a_p))$ liûˋe.\\
    \boxed{2.} Supposons $f$ injective. Soit $(a_1,...,a_p)$ libre. Soit $(\l_1,...,\l_p)\in\K^p$ tels que $\sum_{i=1}^p\l_if(a_i)=0$.\\
    Alors $\sum_{i=1}^p\l_ia_i\in\Ker(f)=\{0\}$ donc $\l_1=...=\l_p=0$ car $(a_1,...,a_p)$ est libre.\\
    \boxed{3.} Supposons $f$ surjective. Soit $(a_1,...,a_p)$ gûˋnûˋratrice. Soit $y\in F$, on a $\exists x \in E, ~ y=f(x)$.\\
    Alors $\exists(\l_1,...,\l_p)\in\K^p, ~ x=\sum_{i=1}^p\l_ia_i$ d'oû¿ $y=\sum_{i=1}^p\l_if(a_i)$, donc $(f(a_1),...,f(a_p))$ est gûˋnûˋratrice.\\
    \boxed{4.} Supposons $f$ bijective, on conclut avec les deux points prûˋcûˋdents.\\
    Rûˋciproquement, soit $(a_1,...,a_p)$ gûˋnûˋratrice de $E$, on a $(f(a_1),...,f(a_p))$ gûˋnûˋratrice de $F$.\\
    Soit $y\in F$, on a $\exists (\l_1,...,\l_p)\in\K^p$ tels que $y=\sum_{i=1}^p\l_if(a_i)$ puis $y=f(\sum_{i=1}^p\l_ia_i)\in\Im(f)$ donc surjective.
\end{prop}

\vspace*{-0.7cm}

\begin{prop}{}{}
    Soient $f\in\L(E,F)$ et $g\in\L(F,G)$.
    \begin{enumerate}
        \item Supposons $g$ bijective, alors $\rg(g\circ f)=\rg(f)$.
        \item Supposons $f$ bijective, alors $\rg(g\circ f)=\rg(g)$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Puisque $g$ est bijective, $g_{|\Im f}$ est bijective de $\Im f$ dans $\Im g_{|\Im f} = \Im(g\circ f)$ donc $\rg(g\circ f) = \rg(f)$.\\
    \boxed{2.} $rg(g\circ f) = \rg(g_{\Im f}) = \rg(g_{|F}) = \rg(g)$
\end{prop}

\begin{defi}{}{}
    Soit $A\in \M_{n,p}(\K)$. On note $f$ l'application linûˋaire canoniquement associûˋe û  $A$ :
    \begin{equation*}
        f : \left\{ \begin{array}{ccc}
            \K^p & \longrightarrow & \K^n \\
            X & \longmapsto & AX
        \end{array}\right.
    \end{equation*}
\end{defi}

\begin{prop}{}{}
    \begin{enumerate}
        \item Soient $A,B\in \M_{n,p}(\K)$, alors $\rg(A+B)\leq\rg(A)+\rg(B)$.
        \item Soient $A\in \M_{n,p}(\K)$, et $B\in\M_{p,q}(\K)$, alors $\rg(AB)\leq \rg(A)$ et $\rg(AB)\leq \rg(B)$.
        \item Soient $A\in \M_{n,p}(\K)$, et $B\in\GL_p(\K)$, alors $\rg(AB)=\rg(A)$.
        \item Soient $A\in \M_{n,p}(\K)$, et $B\in\GL_n(\K)$, alors $\rg(BA)=\rg(B)$.
        \item Soient $A\in \M_{n,p}(\K)$, on a $p=\dim(\Ker A)+\rg(A)$.
    \end{enumerate}
\end{prop}

\begin{ex}{}{}
    Soit $f\in\L(\R^3,\R^2)$, tel que $\forall (x,y,z)\in\R^3, ~ f(x,y,z)=(2x+3z, x+y+z)$. Donner son noyau, image et rang.

    On a donc $\rg(f)=2$ par thûˋorû´me du rang, puis $\Im(f) \subset \R^2$, donc $\Im(f)=\R^2$.
\end{ex}

\section{Endomorphismes.}

\begin{defi}{$\K$-algû´bre.}{}
    Soit $\K$ un corps commutatif, et $A$ un ensemble. Alors $(A,+,\times,\cdot)$ est une $\K$-algû´bre ssi :
    \begin{itemize}
        \item $(A,+,\times)$ est un anneau.
        \item $(A,+,\cdot)$ est un $\K$-espace vectoriel.
        \item $\forall (x,y)\in A^2, ~ \forall \l \in \K \quad \l\cdot(x\times y)=(\l\cdot x) \times y=x\times(\l\cdot y)$.
    \end{itemize}
    La dimension d'une algû´bre est la dimension de l'espace vectoriel sous-jacent.
\end{defi}

\vspace*{-0.7cm}

\begin{prop}{Caractûˋrisation des sous-algû´bres.}{}
    Soit $A$ une $\K$-algû´bre. Soit $B$ un sous-ensemble de $A$. Alors $B$ est une sous-algû´bre de $A$ ssi :
    \begin{itemize}
        \item $1_A\in B$.
        \item $\forall (x,y)\in B^2, ~ \forall \l \in \K, ~ x+\l y \in B$.
        \item $\forall (x,y)\in B^2, ~ x\times y \in B$.
    \end{itemize}
\end{prop}

\vspace*{-0.7cm}

\begin{prop}{}{}
    Montrer que $(\L(E), +, \circ, \cdot)$ est une $\K$-algû´bre.
    \tcblower
    On sait dûˋjû  que $(\L(E),+,\cdot)$ est un $\K$-ev. On va vûˋrifier que $(\L(E),+,\circ)$ est un anneau.\\
    --- $(\L(E),+)$ est bien un groupe abûˋlien car $\K$-ev.\\
    --- La loi $\circ$ est une l.c.i. associative, de neutre $\id_E$.\\
    --- La distributivitûˋ de $\circ$ sur $+$ est vraie û  droite (toujours) et û  gauche (par linûˋaritûˋ).\\
    Enfin, $\forall (f,g)\in\L(E)^2, ~ \forall \l \in \K, ~ \l\cdot(f\circ g)=(\l \cdot f)\circ g=f\circ(\l \cdot g)$.
\end{prop}

\begin{defi}{}{}
    Un projecteur de $E$ est un endomorphisme $p$ de $E$ vûˋrifiant $p\circ p = p$.
\end{defi}

\vspace*{-0.9cm}

\begin{prop}{}{}
    Soit $p$ un projecteur de $E$. Alors $E=\Ker(p)\oplus\Im(p)$, et $\Im(p)=\{x\in E, ~ p(x)=x\}$.
    \tcblower
    On a $p^2=p\ra\Im(p^2)=\Im(p)\et\Ker(p^2)=\Ker(p)\ra E=\Im(p)+\Ker(p)\et\Im(p)\cap\Ker(p)=\{0\}$ (\ref{ex:17}).\\
    \boxed{\subset} Soit $y\in\Im(p)$. On a $\exists x\in E, ~ p(x)=y$ puis $p(y)=p^2(x)=p(x)=y$.\\
    \boxed{\supset} Soit $x\in E$ tel que $p(x)=x$, alors $x\in\Im(p)$.
\end{prop}

\begin{ex}{}{}
    Soit $p\in\L(\R^3)$ tel que $\forall (x,y,z)\in\R^3, ~ p(x,y,z)=(y,y,-x+y+z)$.\\
    Montrer que $p$ est un projecteur et donner ses espaces caractûˋristiques.
    \tcblower
    $\hra$ Dûˋjû , $p$ est une application linûˋaire.\\

    \fbox{Im} $\Im(p)=\{(a,a,b), ~ a,b\in\R\}$.
\end{ex}

\begin{ex}{}{}
    Soient $p,q$ deux projecteurs de $E$.
    \begin{enumerate}
        \item Montrer que $p+q$ est projecteur de $E$ ssi $p\circ q=q\circ p=0$.
        \item Montrer alors que $\Im(p+q)=\Im(p)\oplus\Im(q)$ et $\Ker(p+q)=\Ker(p)\cap\Ker(q)$.
    \end{enumerate}
    \tcblower
    \boxed{1.\ra} Supposons $p+q$ projecteur de $E$. Alors $(p+q)(p+q)-(p+q)=q\circ p+p\circ q=0$ donc $q\circ p=p\circ q=0$.\\
    \boxed{1.\la} Supposons $p\circ q=q\circ p=0$. Alors $(p+q)(p+q)=p+pq+qp+q=p+q$.\\
    \boxed{2.} Soit $y\in\Im(p+q)$, $\exists x \in E, ~ p(x)+q(x)=y$ avec $p(x)\in\Im(p)$ et $q(x)\in\Im(q)$ donc $\Im(p+q)=\Im(p)+\Im(q)$.\\
    Soit $y\in\Im(p)\cap\Im(q)$, $\exists x,x'\in E, ~ y=p(x)=q(x')$ donc $p^2(x)=p\circ q(x')$ donc $p(x)=0$ donc $y=0$.\\
    Par caractûˋrisation, $\Im(p+q)=\Im(p)\oplus\Im(q)$.
\end{ex}

\begin{prop}{Caractûˋrisation des homothûˋties. $\star$}{}
    Soit $f\in\L(E)$. Alors $f$ est une homothûˋtie $\iff \forall x \in E, ~ (x,f(x))$ est une famille liûˋe.
    \tcblower
    Il existe $\l\in\K$ tel que $f=\l\id_E$.\\
    \boxed{\ra} On a bien $(x,\l x)$ liûˋe.\\
    \boxed{\la} Supposons que $\forall x \in E, ~ (x,f(x))$ est liûˋe, donc $\forall x \in E, ~ \exists \l_x \in \K, ~ f(x)=\l_x x$.\\
    Soient $(x,y)\in E^2$, et $\l_x,\l_y$ leurs scalaires associûˋs, on suppose $x,y\neq0_E$.\\
    $\bullet$ Supposons que $(x,y)$ forme une famille libre, on a $f(x)=\l_xx$ et $f(y)=\l_yy$ et $f(x+y)=\l_{x+y}(x+y)$.\\
    On a alors que $\l_xx+\l_yy=\l_{x+y}x+\l_{x+y}y$, et par identification (famille libre), on a $\l_x=\l_{x+y}=\l_y$.\\
    $\bullet$ Supposons que $(x,y)$ soit liûˋe, on a $y=\a x$ avec $\a\in\K^*$, alors $f(y)=\l_yy=\l_y\a x$\\
    On a aussi $f(y)=\a f(x)=(\a\l_x)x$ donc $\l_y\a=\a\l_x$ d'oû¿ $\l_y=\l_x$.\\
    $\bullet$ L'ûˋgalitûˋ est toujours vraie pour $x=0_E$, donc finalement, $f=\l\id_E$. 
\end{prop}

\vspace*{-0.7cm}

\begin{prop}{}{}
    Soit $f\in\L(E)$ tel que $\forall g \in \L(E), ~ f\circ g = g \circ f$, alors $f$ est une homothûˋtie.
    \tcblower
    Soit $x\in E$, si $x=0_E$, alors $(x,f(x))$ est liûˋe.\\
    Sinon, soit $p$ un projecteur sur $\Vect(x)$ parallû´lement û  un supplûˋmentaire de $\Vect(x)$ (existe car dim. finie).\\
    Par hypothû´se, $(f\circ p)(x) = (p \circ f)(x)$ donc $f(x)=p(f(x))$ donc $f(x)\in\Vect(x)$ donc $(x,f(x))$ est liûˋe.\\
    Donc par caractûˋrisation, $f$ est une homothûˋtie.
\end{prop}

\begin{defi}{ûquivalence, similitude.}{}
    Soient $A,B\in \M_{n,p}(\K)$. Elles sont \bf{ûˋquivalentes} ssi :

\end{defi}

        \rg(A)=r\iff A\simeq J_r.
    \end{equation*}
    \tcblower
    \boxed{\la} Supposons $A\simeq J_r$ : $\exists (P,Q) \in \GL_n(\K)\times\GL_p(\K), ~ A = PJ_rQ$.\\
    Or le rang n'est pas modifiûˋ en multipliant par une matrice inversible, donc $\rg(A) = \rg(J_r) = r$.\\
    \boxed{\ra} Notons $f$ l'application linûˋaire canoniquement associûˋe û  $A$.\\
    $\hra$ Soit $(a_1',...,a_r')$ une base de $\Im f$, qu'on complû´te en $\B_2=(a_1',...,a_n')$ une base de $\K^n$.\\
    $\hra$ Puisque $\forall i \in \lb1,r\rb, ~ a_i' \in \Im f$, on a $\exists a_i \in \K^p , ~ a_i' = f(a_i)$.\\
    $\hra$ Par thûˋorû´me du rang, $\dim\Ker f = p-r$, donc on prend $(a_{r+1},...,a_p)$ une base de $\Ker(f)$.\\
    On va vûˋrifier que la famille $\B_1=(a_1,...,a_p)$ est alors une base $\K^p$.\\
    \bf{Libre.} Soit $(\l_1,...,\l_p)\in\K^p$ tels que $\l_1a_1+...+\l_pa_p=0$.\\
    $\hra$ On applique $f$ : $\l_1a_1' + ... + \l_ra_r' = 0$.\\
    $\bullet$ Puisque $(a_1',...,a_r')$ est libre comme base, on a bien $\l_1=...=\l_r=0$.\\
    $\hra$ Il reste $\l_{r+1}a_{r+1}+...+\l_pa_p = 0$, or $(a_{r+1},...,a_p)$ est une base, donc $\l_{r+1}=...=\l_p=0$.\\
    Donc $(a_1,...,a_p)$ est une famille libre de $\K^p$, de cardinal $p$, c'est bien une base de $\K^p$.\\
    On a construit deux bases telles que $\Mat_{\B_2\B_1}(f)=J_r$ donc $A=PJ_rQ$, avec $P=\Mat_{\B_1\B_c}$ et $Q=\Mat_{\B_c\B_2}$.
\end{thm}

\begin{defi}{Forme linûˋaire.}{}
    Soit $E$ un $\K$-ev. Une \bf{forme linûˋaire} sur $E$ est une application linûˋaire de $E$ dans $\K$.\\
    L'ensemble des formes linûˋaires sur $E$ est donc $\L(E,\K)$.
\end{defi}

\vspace*{-0.7cm}

\begin{thm}{}{}
    Soit $E$ un $\K$-ev.
    \begin{enumerate}
        \item Soit $\phi$ une forme linûˋaire et $a\in E$ tel que $\phi(a)\neq0$, alors $E=\Ker(\phi)\oplus\Vect(a)$.
        \item Soient $\phi,\psi$ deux formes linûˋaires non-nulles sur $E$, alors $\Ker(\phi)=\Ker(\psi)\iff\exists \l \in \K, ~ \phi=\l\psi$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Soit $x\in\Ker(\phi)\cap\Vect(a)$, $\exists \a \in \K, ~ x=\a a$, donc $\a\phi(a)=0$ donc $\a=0$, donc $x=0$.\\
    Soit $x\in E$. On a $x=\left( x - \frac{\phi(x)}{\phi(a)}a \right)+\frac{\phi(x)}{\phi(a)}a$. On a bien $E=\Ker(\phi)\oplus\Vect(a)$.\\
    \boxed{2.\la} Supposons $\phi=\l\psi$.\\
    \boxed{\supset} Soit $x\in\Ker(\psi)$, alors $\psi(x)=0$, puis $\phi(x)=0$, donc $x\in\Ker(\phi)$.\\
    \boxed{\subset} Soit $x\in\Ker(\phi)$, alors $\phi(x)=0$, puis $\l \phi(x)=\psi(x)=0$, donc $x\in\Ker(\psi)$.\\
    \boxed{2.\ra} Supposons $\Ker(\phi)=\Ker(\psi)$, et $a\in E$ tel que $\phi(a)\neq0$. On a $E=\Ker(\phi)\oplus\Vect(a)$.\\
    On a de plus $\psi(a)\neq0$ car $\Ker(\phi)=\Ker(\psi)$. On pose $\l=\frac{\phi(a)}{\psi(a)}\in\K^*$.\\
    Soit $x\in E, \exists !(y,\a)\in\Ker(\phi)\times\K, ~ x=y+\a a$, puis $\phi(x)=\a\phi(a)$; $\psi(x)=\a\psi(a)$, donc $\phi(x)=\l(\a\psi(a))$.
\end{thm}

\vspace*{-0.7cm}

\begin{thm}{}{}
    Soit $E$ un $\K$-ev, et $H$ un sous-espace vectoriel de $E$. Alors il y a ûˋquivalence entre :
    \begin{enumerate}
        \item $H$ est le noyau d'une forme linûˋaire non-nulle.
        \item $H$ admet une droite comme supplûˋmentaire.
    \end{enumerate}
    On dit alors que $H$ est un hyperplan de $E$.
    \tcblower
    \boxed{1\ra 2} Fait dans le thûˋorû´me prûˋcûˋdent.\\
    \boxed{2\ra 1} Par hypothû´se, $E=H\oplus\Vect(a)$. Soit $x\in E, ~ \exists!(y,\a)\in\K, ~ x=y+\a a$. On pose $\phi(x)=\a$.\\
    On a bien $\phi:E\to\K$ et linûˋaire (facile) et $H=\Ker(\phi)$. 
\end{thm}

\begin{lemme}{de factorisation.}{}
    Soient $E,F,G$ des $\K$-espaces vectoriels, $f\in\L(E,F)$ et $g\in\L(E,G)$ tels que $\Ker(f)\subset\Ker(g)$.\\
    Alors $\exists h \in \L(F,G)$ telle que $g=h\circ f$.
    \tcblower
    On se place en dimension finie pour simplifier. On note $n=\dim(E)$, $p=\dim(F)$ et $r=\rg(f)$.\\
    Soit $(a'_1,...,a'_r)$ une base de $\Im(f)$ que l'on complû´te en $(a'_1,...,a'_p)$ base de $F$.     On a $\forall i \in \lb1,r\rb, \exists a_i \in E, a'_i=f(a_i)$.\\
    On dûˋfinit $h$ comme l'unique application linûˋaire de $F$ dans $G$ telle que $\forall i \in \lb1,r\rb, ~ h(a'_i)=g(a_i)$.\\
    Il reste û  vûˋrifier que $g=h\circ f$. On cherche une base de $E$ utilisant une base de $\Ker(f)$ : $(a_{r+1},...,a_n)$.\\
    Notons $\B=(a_1,...,a_r,a_{r+1},...,a_n)$ base de $E$. Vûˋrifions enfin que $\forall i \in \lb1,n\rb, ~ g(a_i)=h\circ f(a_i)$.\\
    Si $i\in\lb1,r\rb$, alors $(h\circ f)(a_i)=h(a'_i)=g(a_i)$ par dûˋfinition de $h$.\\
    Sinon, $i\in\lb r+1,n\rb$, donc $(h\circ f)(a_i)=h(0)=0$ et $g(a_i)=0$ car $\Ker(f)\subset\Ker(g)$.
\end{lemme}

\section{Sommes directes.}

\begin{rappel}{}{}
    Soit $E$ un $\K$-ev, et $F$ et $G$ deux sous-espaces vectoriels de $E$.\\
    --- $F$ et $G$ sont en somme directe, notûˋe $F\oplus G$, ssi $F\cap G=\{0\}$.\\
    --- $F$ et $G$ sont supplûˋmentaires dans $E$ ssi $E=F\oplus G$.\\
    --- $E$ est somme directe de $F$ et $G$ ssi $\forall x \in E, ~ \exists !(y,z)\in F\times G, ~ x=y+z$.\\
    --- Si $E$ est somme directe de $F$ et $G$, on dûˋfinit $p$ projecteur sur $F$, parallû´lement û  $G$, $q$ ...:
    \begin{equation*}
        \forall x \in E, ~ p(x)=y \text{ et } q(x)=z, ~ p+q=\id_E.
    \end{equation*}
    Alors $\Im(p)=F$, et $\Ker(p)=G$; $\Ker(q)=F$ et $\Im(q)=G$.
\end{rappel}

\vspace*{-0.8cm}

\begin{defi}{Somme directe de $n$ espaces. $\star$}{}
    Soit $E$ un $\K$-ev et $(E_1,...,E_n)$ des sous-espaces vectoriels de $E$.
    \begin{equation*}
        E_1 \oplus ... \oplus E_{n+1} = (E_1 \oplus ... \oplus E_n) \oplus E_{n+1}.
    \end{equation*}
    \warning Cette dûˋfinition est plus forte que la somme directe deux-û -deux.
\end{defi}

\vspace*{-0.8cm}

\begin{prop}{Caractûˋrisation de la somme directe.}{}
    Soient $E_1,...,E_n$ des ssev de $E$. Alors les propositions suivantes sont ûˋquivalentes :
    \begin{enumerate}
        \item $E_1,...,E_n$ sont en somme directe.
        \item $\forall (x_1,...,x_n),(y_1,...,y_p)\in E_1\times...\times E_n, ~ x_1+...+x_n=y_1+...+y_n \ra \forall i \in \lb1,n\rb, ~ x_i=y_i$.
        \item $\forall (x_1,...,x_n)\in E_1\times...\times E_n, ~ x_1+...+x_n=0_E \ra \forall i \in \lb1,n\rb, ~ x_i = 0_E$.
    \end{enumerate}
    \tcblower
    \boxed{2\ra 3} Soient $x_1,...,x_n\in E_1\times...\times E_n$ tels que $x_1+...+x_n=0_E$. Alors $x_1+...+x_n=0_E+...+0_E$.\\
    Alors d'aprû´s l'hypothû´se, $\forall i \in \lb1,n\rb, ~ x_i=0_E$.\\
    \boxed{3\ra 2} Soient $(x_1,...,x_n),(y_1,...,y_n)\in E_1\times...\times E_n$ tels que $x_1+...+x_n=y_1+...+y_n$.\\
    Alors $(x_1-y_1)+...+(x_n-y_n)=0_E$, puis par hypothû´se $\forall i \in \lb1,n\rb, ~ x_i-y_i=0_E$, donc $x_i=y_i$.\\
    \boxed{1\ra3} Soient $x_1,...,x_n\in E_1\times...\times E_n$ tels que $x_1+...+x_n=0_E$, alors $x_n=-x_1-...-x_{n-1}$.\\
    Alors $x_n\in E_n \cap (E_1+...+E_{n-1})=\{0_E\}$ donc $x_n = 0_E$. On rûˋitû´re avec $x_1+...+x_{n-1}=0_E$.\\
    \boxed{3\ra1} Soit $x\in E_n\cap (E_1+...+E_{n-1})$. Alors $\exists (x_1,...,x_{n-1})\in E_1\times...\times E_{n-1}, ~ x = x_1+...+x_{n-1}$.\\
    Alors $x_1+...+x_{n-1}-x=0$, ils sont donc tous nuls par hypothû´se. On rûˋitû´re sur $E_1+...+E_{n-1}$.
\end{prop}

\vspace*{-0.8cm}

\begin{defi}{Projecteurs associûˋs û  une somme directe. $\star$}{}
    Soit $E$ un $\K$-ev, et $E_1,...,E_n$ des ssev de $E$ tels que $E=E_1\oplus...\oplus E_n$.\\
    En particulier, $\forall x \in E, ~ \exists!(x_1,...,x_n)\in E_1\times...\times E_n, ~ x=x_1+...+x_n$.\\
    On dûˋfinit $p_i:x\mapsto x_i$ pour tout $i\in\lb1,n\rb$.
    \tcblower
    On vûˋrifie que les $p_i$ sont des projecteurs. Soient $x,y\in E$ et $\l\in\K$, $x=x_1+...+x_n$ et $y=y_1+...+y_n$.\\
    On a alors $x+\l y = (x_1+\l y_1)+...+(x_n+\l y_n)$, puis par unicitûˋ, $p_i(x+\l y)=x_i+\l y_i = p_i(x)+\l p_i(y)$.\\
    On a de plus que $p_i(x)=x_i$, et $p_i(x_i)=x_i$ donc $p_i\circ p_i = p_i$.\\
    On a aussi $\Ker(p_i)=\bigoplus_{j\neq i}E_j$, et $\Im(p_i)=E_i$, et $\sum_ip_i=\id_E$, et $p_i\circ p_j = 0$ pour $i\neq j$.
\end{defi}

\begin{thm}{$\star$}{}
    Soit $E$ un $\K$-evdf et $E_1,...E_n$ des ssev de $E$.\\
    Alors $E=E_1\oplus...\oplus E_n$ ssi on obtient une base de $E$ en concatûˋnant des bases de $E_1,...,E_n$.
\end{thm}

\begin{prop}{$\star$}{}
    Soit $E$ un $\K$-evdf et $E_1,...,E_n$ des sev de dimensions finies.
    \begin{equation*}
        \dim(E_1+...+E_n) \leq \dim(E_1) + ... + \dim(E_n).
    \end{equation*}
    \tcblower
    Par rûˋcurrence sur $n$ :\\
    \bf{Initialisation.} $\dim(E_1+E_2)=\dim(E_1)+\dim(E_2)-\dim(E_1\cap E_2) \leq \dim(E_1)+\dim(E_2)$ (GRASSMAN).\\
    \bf{Hûˋrûˋditûˋ.} Soit $n\in\N$ tel que la propriûˋtûˋ soit vraie aux rangs $2$ et $n$. Alors :
    \begin{equation*}
        \dim(E_1+...+E_n+E_{n+1})\leq \dim(E_1+...+E_n)+\dim(E_{n+1})\leq \sum_{i=1}^n\dim(E_i) + \dim(E_{n+1}).
    \end{equation*}
    Par rûˋcurrence, on a la propriûˋtûˋ pour tout $n\in\N$.
\end{prop}

\begin{thm}{Caractûˋrisation des sommes directes en dimension finie. $\star$}{}
    Soit $E$ un $\K$-evdf et $E_1,...,E_n$ des ssev de dimensions finies.
    \begin{equation*}
        E_1\oplus...\oplus E_n \iff \dim(E_1+...+E_n)=\dim(E_1)+...+\dim(E_n).
    \end{equation*}
    \tcblower
    \boxed{\ra} Mûˆme rûˋcurrence que prûˋcûˋdemment.\\
    \boxed{\la} Par rûˋcurrence sur $n$ :\\
    \bf{Initialisation.} $\dim(E_1+E_2)=\dim(E_1)+\dim(E_1)$ donc $\dim(E_1\cap E_2)=0$, donc $E_1\oplus E_2$.\\
    \bf{Hûˋrûˋditûˋ.} Soit $n\in\N$ tel que la propriûˋtûˋ soit vraie aux rangs $2$ et $n$. Alors :
    \begin{equation*}
        \dim((E_1+...+E_n)+E_{n+1}) \leq \dim(E_1+...+E_n) + \dim(E_{n+1}) \leq \dim(E_1) + ... + \dim(E_{n+1}).\\
    \end{equation*}
    On a \small$\dim(E_1+...+E_{n+1})=\dim(E_1)+...+\dim(E_{n+1})$ donc $\dim((E_1+...+E_n)+E_{n+1})=\dim(E_1+...+E_n)+\dim(E_{n+1})$.\\
    Donc d'aprû´s $P(2)$, $E_{n+1}\oplus(E_1+...+E_n)$ puis $\dim(E_1+...+E_n)=\dim(E_1)+...+\dim(E_n)$ donc $E_1\oplus...\oplus E_n$.\\
    On conclut que $E_1\oplus ... \oplus E_{n+1}$.
\end{thm}

\section{Sous-espaces stables.}

\begin{defi}{Espace stable.}{}
    Soit $E$ un $\K$-ev, $F$ un ssev de $E$ et $u\in\L(E)$.
    \begin{center}
        $F$ est \bf{stable} par $u$ ssi $u(F)\subset F$.
    \end{center}
\end{defi}

\begin{defi}{Endomorphisme induit.}{}
    Soit $F$ un ssev de $E$ stable par $u\in\L(E)$. On pose :
    \begin{equation*}
        \tilde{u} : \begin{cases}
            F & \to \quad F\\
            x & \mapsto \quad u(x)
        \end{cases}
    \end{equation*}
    On appelle $\tilde{u}\in\L(F)$ l'endomorphisme induit par $u$ sur $F$.
\end{defi}

\vspace*{-0.4cm}

\begin{prop}{$\star$}{}
    Soit $E$ un $\K$-ev et $u,v\in\L(E)$ tels que $uv=vu$, alors $\Im(u)$ et $\Ker(u)$ sont stables par $v$.
    \tcblower
    $\bullet$ Soit $x\in\Ker(u)$. Alors $uv(x)=vu(x)=v(0_E)=0_E$ donc $v(x)\in\Ker(u)$.\\
    $\bullet$ Soit $z\in v(\Im u)$, $\exists y \in \Im(u), ~ z=v(y)$ puis $\exists x \in E, ~ y=u(x)$, donc $z=vu(x)=uv(x)\in\Im(u)$.
\end{prop}

\vspace*{-0.4cm}

\begin{corr}{$\star$}{}
    Soit $E$ un $\K$-ev et $u\in \L(E)$, alors $\forall k \in \N, ~ F_k=\Im(u^k)$ et $G_k=\Ker(u^k)$ sont stables par $u$.
    \tcblower
    En effet, $\forall k \in \N, ~ u^k\circ u = u \circ u^k = u^{k+1}$.
\end{corr}

\vspace*{-0.4cm}

\begin{exi}{$\star$}{}
    Soit $E$ un $\K$-ev. Quels sont les endomorphismes de $E$ qui laissent toutes les droites stables ?
    \tcblower
    \bf{Analyse.} Soit $u\in\L(E)$ laissant toutes les droites stables. Soit $x\in E$ non nul. On prend la droite $\m{D}=\Vect(x)$.\\
    Alors par hypothû´se, $u(\m{D})\subset\m{D}$, donc $u(x)\in\m{D}$, donc $\exists \l \in \K, ~ u(x)=\l x$, donc $u$ est une homothûˋtie.\\
    \bf{Synthû´se.} $\forall \l \in \K, ~ \l\id_E$ laisse toutes les droites stables.
\end{exi}

\begin{thm}{Caractûˋrisation des sous-espaces stables en dimension finie. $\star$}{}
    Soit $E$ un $\K$-evdf de dimension $n$, soit $F$ un ssev de $E$, et $u\in\L(E)$.
    \begin{center}
        $F$ est stable par $u$ ssi il existe une base $\B$ de $E$ telle que $\Mat_{\B}(u)=\begin{pmatrix}A_1&A_2\\0&A_3\end{pmatrix}$.
    \end{center}
    Avec $A_1\in M_{p}(\K)$, $A_2\in M_{p,n-p}(\K)$ et $A_3\in M_{n-p,p}(\K)$.
    \tcblower
    \boxed{\ra} Supposons $F$ stable par $u$. Notons $\tilde{u}$ la restriction de $u$ û  $F$.\\
    Soit $\B_1=(e_1,...,e_p)$ une base de $F$. C'est une famille libre de $E$, on la complû´te en $\B=(e_1,...,e_n)$ base de $E$. Alors $G=\Vect(e_{p+1},...,e_n)$ est \bf{un} supplûˋmentaire de $F$. Alors :
    \begin{equation*}
        \Mat_{\B}(u)=\begin{pmatrix}A_1&\Big|&A_2\\\hline(0)&\Big|&A_3\end{pmatrix}
    \end{equation*}
    En effet, le $(0)$ vient de la stabilitûˋ de $F$ par $u$ : $\forall j \in \lb1,p\rb, ~ u(e_j)\in\Vect(e_1,...,e_p)$.\\
    \boxed{\la} Triviale.
\end{thm}

\begin{exercice}{}{}

\end{exercice}
