Dans tout le chapitre, $E$ et $F$ sont des $\R$-evdf. On rappelle quelques résultats utiles en \bf{dimension finie} :\\
$\hra$ Toutes les normes sont équivalentes.\\
$\hra$ $K$ est un compact ssi il est fermé et borné.\\
$\hra$ L'image continue d'un compact est compacte.\\
$\hra$ Une fonction continue sur un compact est bornée et atteint ses bornes.\\
$\hra$ Les applications multi-linéaires sont continues, donc aussi lipschitziennes.\\
$\hra$ Les applications coordonnées sont linéaires donc continues et lipschitziennes.\\
$\hra$ Tous les polynômes à plusieurs variables sont continus.

\begin{exercice}{}{}
    \begin{enumerate}
        \item Montrer que $f(x,y)=\frac{xy}{\sqrt{x^2+y^2}}$, $f(0,0)=0$ est continue sur $\R^2$.
        \item Montrer que $f(x,y)=\frac{xy}{x^2+y^2}$ n'est pas prolongeable par continuité en $(0,0)$.
    \end{enumerate}
    \tcblower
    \boxed{1.} $\R^2\setminus\{0,0\}$ est un ouvert de $\R^2$, donc $f$ y est continue par théorème.\\
    En $(0,0)$ : $\forall (x,y) \in \R^2, ~ |xy|\leq\frac{1}{2}(x^2+y^2)$ donc $|f(x,y)|\leq\frac{1}{2}\sqrt{x^2+y^2}\xrightarrow[(x,y)\to(0,0)]{}0$.\\
    \boxed{2.} Supposons que $f$ admette une limite en $(0,0)$, alors $\lim_{x\to0}f(x,0)=0$, si $l$ existe alors $l=0$.\\
    Posons $\forall n \geq 1, ~ u_n=v_n=\frac{1}{n}$, $(u_n,v_n)\to(0,0)$ mais $f(u_n, v_n) = \frac{u_nv_n}{u_n^2+v_n^2}=\frac{1}{2}\not\to0$.\\
    Donc $f$ n'admet pas de limite en $(0,0)$.
\end{exercice}

\section{Dérivée suivant un vecteur et dérivées partielles.}

\begin{defi}{Dérivée suivant un vecteur.}{}
    Soient $E,F$ deux $\R$-evdf, $f:U\subset E \to F$ avec $U$ un ouvert de $E$, $a\in U$ et $v\in E$ non-nul.\\
    On dit que $f$ admet une dérivée de $a$ selon le vecteur $v$ si et seulement si :
    \begin{equation*}
        \frac{f(a+tv) - f(a)}{t} \xrightarrow[t\to0]{}D_vf(a) \in F.
    \end{equation*}
    Cette limite, si elle existe, est appelée dérivée de $f$ en $a$ suivant le vecteur $v$.
\end{defi}

\begin{defi}{Dérivées partielles.}{}
    Soient $E,F$ des $\R$-evdf, $U$ un ouvert de $E$, $f:U\to E$, $a\in U$, $n$ la dimension de $E$ et $\B=(e_1,...,e_n)$ base de $E$.\\
    Les dérivées partielles de $f$ en $a$ sont les vecteurs dérivées de $f$ en $a$ suivant les vecteurs de la base $\B$.\\
    On les notera $\6_if(a)$ pour $i\in\lb1,n\rb$ :
    \begin{equation*}
        \forall i \in \lb1,n\rb, ~ \6_if(a) =\lim_{t\to0}\frac{f(a+te_i)-f(a)}{t} ~ \nt{si bien défini.} 
    \end{equation*}
\end{defi}

\vspace*{-0.8cm}

\begin{defi}{}{}
    Soit $f:U\subset E \to F$. $f$ admet des dérivées partielles sur $U$ ssi $\forall a \in U$, $f$ admet des dérivées partielles en $a$.
\end{defi}

\vspace*{-0.8cm}

\begin{ex}{}{}
    Soit $f(x,y)=x^2+6xe^y - 3y^6$ pour $(x,y)\in\R^2$ on prend implicitement la base canonique de $\R^2$.
    \begin{equation*}
        \frac{\6 f}{\6 x}(x,y) = D_{(1,0)}f(x,y) = 2x + 6e^y; \quad \frac{\6f}{\6y}(x,y) = D_{(0,1)}f(x,y) = 6xe^y - 18y^5.
    \end{equation*}
\end{ex}

\vspace*{-0.8cm}

\begin{prop}{Règles de calcul.}{}
    \begin{enumerate}
        \item $\ds \forall i \in \lb1,n\rb ~ : ~\frac{\6}{\6x_i}(f+\l g) = \frac{\6}{\6x_i} f + \l \frac{\6}{\6x_i}g$.
        \item Si $f$ et $g$ sont à valeurs réelles, $\ds\forall i \in \lb1,n\rb ~:~ \frac{\6}{\6 x_i}(fg) = f\frac{\6}{\6x_i}g+g\frac{\6}{\6x_i}f$.
    \end{enumerate}
    Il suffit d'utiliser $\frac{\6 f}{\6 x_i}(a)=\phi_i'(a)$ et les règles usuelles de dérivation.
\end{prop}

\vspace*{-0.8cm}

\begin{exercice}{}{}
    \begin{align*}
        \begin{cases}
            f(x,y) = \frac{xy}{x^2+y^2} & \text{si } (x,y)\neq(0,0) \\
            f(0,0) = 0
        \end{cases}
    \end{align*}
    Montrer que $f$ admet des dérivées partielles sur $\R^2$ mais que $f$ n'est pas continue sur $\R^2$.
    \tcblower
    On a déjà vu que $f$ n'est pas continue en $(0,0)$.
    \begin{equation*}
        \lim_{n\to+\infty} f\left( \frac{1}{n}, \frac{1}{n} \right) = \frac{1}{2} \neq f(0,0) = 0.
    \end{equation*}
    Sur l'ouvert $\R^2\setminus\{0,0\}$, $f$ admet des dérivées partielles par les théorèmes généraux.
    \begin{equation*}
        \frac{\6 f}{\6 x}(x,y) = \frac{y(x^2+y^2)-xy(2x)}{(x^2+y^2)} = \frac{y^3 - x^2y}{(x^2+y^2)^2}
    \end{equation*}
    Par symétrie des lettres, $\frac{\6 f}{\6 y}(x,y) = \frac{x^3 - y^2x}{(x^2+y^2)^2}$.\\
    En $(0,0)$, on a :
    \begin{equation*}
        \frac{\6 f}{\6 y}(0,0) = \frac{\6 f}{\6 x}(0,0) = \lim_{h\to 0}\frac{\cancel{f(h,0)} - \cancel{f(0,0)}}{h} = 0
    \end{equation*}
    Donc $f$ admet des dérivées partielles sur $\R^2$ tout entier.
\end{exercice}

\vspace*{-0.5cm}

Les dérivées partielles de $f$ en $a$ donnent des contraintes sur les valeurs de $f$ autour de $a$, mais uniquement dans la direction des vecteurs de la base alors que $f$ continue en $a$ nécessite de contrôler les valeurs de $f$ sur tout un voisinage de $a$. L'existence des dérivées partielles ne garantit pas la continuité. 

\begin{exercice}{}{}
    Pour tout $M\in\M_n(\R)$, on pose $f(M)=\det(M)$. Choisissons la base $\B=(E_{i,j})$ des matrices élémentaires.\\
    Montrer que $f$ admet des dérivées partielles sur $\M_n(\R)$ et les calculer.
    \tcblower
    Soit $A\in\M_n(\R)$. Notons $(C_1,...,C_n)$ les colonnes de $A$.
    \begin{align*}
        f(A+t E_{i,j}) - f(A) &= \det(C_1,...,C_j + t e_i, ..., C_n) - \det(C_1,...,C_n)\\
        &=\cancel{\det(C_1,...,C_j, ..., C_n) - \det(C_1,...,C_n)} + t\det(C_1,...,C_{j-1}, e_i, C_{j+1}, ..., C_n)
    \end{align*}
    Donc $\forall t \neq 0$ : 
    \begin{equation*}
        \frac{f(A+tE_{i,j})-f(A)}{t} = \det(C_1,...,C_{j-1}, e_i, C_{j+1}, C_n)
    \end{equation*}
    Donc $f$ admet une dérivée partielle en $A$ suivant $E_{i,j}$ et $D_{E_{i,j}}f(A)=(-1)^{i+j}\D_{i,j}=[\Com A]_{i,j}$.
\end{exercice}

\begin{defi}{Point critique.}{}
    Soit $f:U\subset E \to F$ admettant des dérivées partielles en $a\in U$ avec $\B$ base de $E$.\\
    On dit que $a$ est un \bf{point critique} de $f$ ssi $\forall i \in \lb1,n\rb, ~ \6_if(a)=0$.
\end{defi}

\vspace*{-0.8cm}

\begin{prop}{}{}
    Soit $f:U\subset E \to \R$ admettant des dérivées partielles en $a\in U$ avec $\B$ base de $E$.\\
    Dans ces hypothèses, si $a$ est un extremum local, alors c'est un point critique.
    \tcblower
    Soit $a$ un extremum local de $f$, on peut supposer que c'est un maximum local sans perte de généralité.
    \begin{equation*}
        \exists V \in \cursive{V}(a), ~ \forall x \in V, ~ f(x) \leq f(a).
    \end{equation*}
    Soit $i\in\lb1,n\rb$, on identifie $E$ à $\R^n$, où $n=\dim E$, et $x\in E$ à $(x_1,...,x_n)$ dans la base $\B$.\\
    On pose $a=(a_1,...,a_n)$ et $\forall t \in \cursive{V}(a_i), ~ \phi_i(t)=f(a_1,...,a_{i-1},t,a_{i+1},a_n)$.\\
    Par choix de $a$, $\phi_i$ admet un extremum local en $a_i$, or $\phi_i$ est dérivable en $a_i$ et $\phi'_i(a_i)=\frac{\6 f}{\6 x_i}(a)$.\\
    Par théorème de Rolle, on sait d'ailleurs que $\phi'_i(a_i)=0$ donc $\forall i \in \lb1,n\rb, ~ \frac{\6 f}{\6 x_i}(a)=0$.
\end{prop}

\vspace*{-0.8cm}

\begin{ex}{}{}
    On pose pour $(x,y)\in\R^2 ~:~ f(x,y)=x^2-y^2$. Chercher les points critiques de $f$ sur $\R^2$.
    \tcblower
    \begin{equation*}
        \begin{cases}\frac{\6 f}{\6 x}(x,y) = 0  \\ \frac{\6 f}{\6 y} = 0 \end{cases} \iff \begin{cases}
            2x = 0 \\ -2y = 0 
        \end{cases} \iff (x,y) = (0,0)
    \end{equation*}
    Donc $(0,0)$ est le seul point critique de $f$, mais
    \begin{equation*}
        \begin{cases}
            \forall n \in \N^*, ~ f\left( \frac{1}{n}, 0 \right) = \frac{1}{n^2} > 0 \\ \forall n \in \N^*, ~ f\left( 0, \frac{1}{n} \right) = -\frac{1}{n^2} < 0 
        \end{cases} \nt{ donc $(0,0)$ n'est pas un extremum local.}
    \end{equation*}
\end{ex}

\begin{prop}{Fonctions coercives.}{}
    Soit $f:E\to\R$ avec $E$ un $\R$-evdf. On suppose $f$ continue sur $E$ et $\lim_{\|x\|\to+\infty}f(x)=+\infty$.\\
    Alors $f$ admet un minimum sur $E$.
    \tcblower
    Par hypothèse, $\forall A \in \R, ~ \exists B \in \R, ~ \forall x \in E, ~ \|x\|>  B \ra f(x)\geq A$.\\
    Prenons $A=f(0)$ : $\exists R \in \R, ~ \forall x \in E, ~ \|x\|> R\ra f(x)\geq f(0)$.\\
    Puisque $f$ est continue sur $E$, $f$ est continue sur $B_f(0,R)$, qui est compact comme fermé borné en dim. finie.\\
    Ainsi, $f$ est continue sur le compact $B_f(0,R)$ et à valeurs réelles donc est minorée sur $B_f(0,R)$ et atteint son minimum $m$. Donc $\exists x_0 \in B_f(0,R), ~ m = f(x_0)$. Soit $x\in E$.\\
    $\hra$ Si $x\in B_f(0,R)$, alors $f(x) \geq m = f(x_0)$ par choix de $m$/\\
    $\hra$ Sinon, $\|x\|>R$ et $f(x)\geq f(0)\geq m$.
\end{prop}

\vspace*{-0.3cm}

\begin{exercice}{}{}
    Soit $f(x,y)=x^4+8xy+y^2-4y+6$ pour $x,y\in\R$.
    \begin{enumerate}
        \item $f$ est-elle bornée sur $[0,1]^2$ ? Déterminer ses bornes.
    \end{enumerate}
    \tcblower
    \boxed{1.} On a $K=[0,1]^2$ compact de $\R^2$ comme produit de compacts, donc $f$ est continue sur $K$ et à valeurs réelles donc $f$ est bornée sur $K$ et atteint ses bornes. Si le max est atteint sur $]0,1[^2$, alors il est atteint en un point critique $a=(a_1,a_2)$.\\
    \bf{Points critiques.} On a $4a_1^3 + 8a_2 = 0$ et $8a_1 + 2a_2 - 4 = 0$ donc $a_2=-\frac{a_1^3}{2}<0$, il n'y a pas de point critique.\\
    Le maximum est atteint aux bords : on étudie les valeurs de $f$ sur les bords de $K$.\\
    On étudie $\phi_1:y\mapsto f(0,y)$; $\phi_2:x\mapsto f(x,0)$; $\phi_3:y\mapsto f(1,y)$ et $\phi_4:x\mapsto f(x,1)$ pour $(x,y)\in[0,1]^2$.
\end{exercice}

\section{Différentielle.}

\begin{defi}{}{}
    Soit $f:U\subset E\to F$ et $a\in U$. Alors $f$ est différentiable en $a$ ssi $f$ admet un DL1 en $a$ ssi :
    \begin{equation*}
        \exists L \in \L(E,F], ~ f(a+h) =f(a) + L(h) + o(\|h\|).
    \end{equation*}
    \bf{Remarque.} $L\in\L(E,F)$ avec $E$ de dimension finie, donc $L$ est continue sur $E$.
\end{defi}

\vspace*{-0.3cm}

\begin{lemme}{}{}
    Soient $L_1,L_2\in\L(E,F)$ égales sur un voisinage de zéro, alors $L_1=L_2$.\\
    Soit $L\in\L(E,F)$. Si $L(h)=o(\|h\|)$, alors $L=0$.
    \tcblower
    \boxed{1.} Par hypothèse, il existe $r>0, ~ \forall h \in B(0,r), ~ L_1(h)=L_2(h)$. Soit $x\in E$.\\
    $\hra$ Si $x=0$, alors $L_1(0)=L_2(0)$ par hypothèse.\\
    $\hra$ Sinon, on a $\frac{r}{2\|x\|}x\in B(0,r)$, donc $L_1(\frac{r}{2\|x\|}x)=L_2(\frac{r}{2\|x\|}x)$ puis $\frac{r}{2\|x\|}L_1(x)=\frac{r}{2\|x\|}L_2(x)$ puis $L_1(x)=L_2(x)$.\\
    \boxed{2.} Par hypothèse, $L(h)=\|h\|\e(h)$ avec $\e(h)\to0$. Soit $x\in E$.\\
    $\hra$ Si $x=0$, alors $L(tx)=0$.\\
    $\hra$ Sinon, pour $t\in\R^*$, $L(tx)=\|tx\|\e(tx)$ avec $\e(tx)\to0$ puis $tL(x)=|t|\|x\|\e(tx)$ puis $L(x)=\frac{|t|}{t}\|x\|\e(tx)\to0$.\\
    Par passage à la limite, $L(x)=0_E$.
\end{lemme}

\begin{defi}{Différentielle de $f$ en $a$.}{}
    Soit $f:U\subset E\to F$ et $a\in U$. On suppose $f$ différentiable en $a$ :
    \begin{equation*}
        \exists L \in \L(E,F), ~ f(a+h) = f(a) + L(h) + \|h\|\e(h) \nt{ avec } \e(h)\xrightarrow[h\to0_E]{}0.
    \end{equation*}
    Alors $L$ est déterminée de manière unique, et notée $df_a$, appelée différentielle de $f$ en $a$.\\
    \bf{Remarque.} L'unicité de $L$ est acquise par le lemme précédent.
\end{defi}

\begin{defi}{Différentielle de $f$ sur $U$.}{}
    Soit $f:U\subset E \to F$. Si $f$ est différentiable en $a$ pour tout $a\in U$, alors $f$ est différentiable sur $U$ et :
    \begin{equation*}
        df : \begin{cases} U &\to \quad \L(E,F) \\ a &\mapsto \quad df_a\end{cases}.
    \end{equation*}
    est la différentielle de $f$ sur $U$.
\end{defi}

\begin{ex}{Exemples de calculs de différentielles.}{}
    \begin{enumerate}
        \item Si $f$ est constante sur $U$, alors $f(a+h)-f(a)=0=L(h)+\|h\|\e (h)$ donc $L=0$ i.e. $df_a=0$.
        \item Pour $f$ linéaire et $a\in U$, $f(a+h)-f(a)=f(h)=L(h)+o(\|h\|)$.\\
        Alors $L(h)=f(h)$ convient, donc $L=df_a(h)=f(h)$, $df_a=f$.
        \item Soit $a\in U\subset \R$, $U$ ouvert de $\R$ et $f:U\to\R$.\\
        On a $f$ différentiable en $a$ ssi $f$ admet un DL à l'ordre 1 en $a$ : $f(a+h)=f(a)+L(h)+o(h)$.\\
        Alors $f$ est différentiable en $a$ ssi $f$ est dérivable en $a$ et $f'(a)=df_a(1)$.
        \item Posons $f(M)=M^2$ pour $M\in\M_n(\R)$. Soit $A\in\M_n(\R)$.\\
        On a $f(A+H)-f(A)=(A+H)^2-A^2 = A^2+H^2+AH+HA-A^2=AH+HA+H^2=L(H)+o(\|H\|)$.\\
        Avec $L(H)=AH+HA$ et $\e(H)=\frac{H^2}{\|H\|}$ pour $H\neq0$ avec $\|\.\|$ une norme d'algèbre.\\
        On a en effet $L$ linéaire et $\|\e(H)\|=\frac{\|H^2\|}{\|H\|}\leq\frac{\|H\|^2}{\|H\|}=\|H\|\to0$ donc $\e(H)\to0$. 
        \item Soit $E$ euclidien, $f(x,y)=\Lg x,y\Rg$, soient $a,b\in E$.\\
        On a $f((a,b)+(h,k))=f(a+h,b+k)=\Lg a, b \Rg + \Lg h,b\Rg + \Lg a,k\Rg + \Lg h,k\Rg = f(a,b) + L((h,k)) + o(\|(h,k)\|)$.\\
        Avec $L(h,k)=\Lg a,k \Rg + \Lg h , b\Rg$ linéaire, et par C.S. $|\Lg h,k\Rg|\leq \|h\|\|k\|\leq \|(h,k)\|^2$ avec $\|\.\|_E$ euclidienne, et avec $\| (x,y) \|=\max(\|x\|_E,\|y\|_E)$; alors $\Lg h,k\Rg = o(\|(h,k)\|)$.
    \end{enumerate}
\end{ex}

\begin{exercice}{}{}
    Soit $E$ un $\R$-evdf, $\|\.\|$ une norme sur $E$ et $f:x\mapsto\|x\|$. Montrer que $f$ n'est pas différentiable en 0.
    \tcblower
    Supposons $f$ différentiable en $0$, alors :
    \begin{equation*}
        \exists L \in \L(E,F), ~ f(h) = L(h) + \|h\|\e(h).
    \end{equation*}
    Soit $h\in E$ fixé et $t\in\R$, alors $\|th\|=L(th)+\|th\|\e(th)$ puis $|t|\|h\|=t L(h)+|t|\|h\|\e(th)$.\\
    Alors pour $t\neq0$, $\|h\|=\frac{t}{|t|}L(h)+\|h\|\e(th)$ puis par passage à la limite en $0^+$ : $\|h\|=L(h)$; en $0^-$ : $\|h\|=-L(h)$.\\
    Donc $\forall h \in E, ~ L(h)=0$, puis $\|h\|=\|h\|\e(h)$, absurde.
\end{exercice}

\begin{prop}{}{}
    Soit $f:U\subset E \to F$ et $a\in U$.
    \begin{enumerate}
        \item Si $f$ est différentiable en $a$, alors $f$ est continue en $a$.
        \item Si $f$ est différentiable sur $U$, alors $f$ est continue sur $U$.
    \end{enumerate}
    \tcblower
    \boxed{1.} $f(a+h)=f(a)+L(h)+\|h\|\e(h)$ avec $L=df_a\in\L(E,F)$ et $\e(h)\to0$.\\
    Puis $L$ est continue sur $E$ donc $L(h)\xrightarrow[h\to 0]{}L(0)=0$ et $\|h\|\e(h)\xrightarrow[h\to0]{}0$ donc $f(a+h)\xrightarrow[h\to0]{} f(a)$.
\end{prop}

\begin{prop}{}{}
    Soit $f:U\subset E\to F$, avec $E$ de dimension $n$ et $\B=(e_1,...,e_n)$ base de $E$. Soit $a\in U$ et $v\in E\setminus\{0_E\}$.\\
    Si $f$ est différentiable en $a$, alors $f$ admet une dérivée en $a$ suivant le vecteur $v$ et $Df_v(a)=df_a(v)$.\\
    En particulier, si $f$ est différentiable en $a$, alors $f$ admet des dérivées partielles en $a$.
    \tcblower
    Supposons que $f$ soit différentiable en $a$, alors $f(a+h)=f(a)+L(h)+\|h\|\e(h)$ avec $L=df_a$ et $\e(h)\to0$.\\
    Soit $t\in\R$. On a $f(a+tv)=f(a)+L(tv)+\|tv\|\e(tv)$ donc $f(a+tv)-f(a)=L(tv)+\|tv\|\e(tv)$.\\
    Puis $\forall t \in \R^*, ~ \frac{f(a+tv)-f(a)}{t}=L(v)+\|v\|\e(tv)\xrightarrow[t\to0]{} L(v)$.\\
    Donc $Df_v(a)=L(v)=df_a(v)$, c'est vrai quel que soit $v$, donc $f$ admet des dérivées partielles.
\end{prop}

\begin{exercice}{}{}
    \begin{enumerate}
        \item Soit $(\A,+,\times,\.)$ une $\K$-algèbre normée et $u\in\A$ tel que $\|u\|<1$. Notons $e$ le neutre de $\A$ pour $\times$.
        Montrer que $e-u$ est inversible et que $(e-u)^{-1}=\sum_{n=0}^\infty u^n$.
        \item On pose $f:A\mapsto A^{-1}$. Montrer que $f$ est diff. sur l'ouvert $\GL_n(\R)$ et déterminer $df_A$ pour tout $A\in\GL_n(\R)$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Vérifions déjà que $\sum_{n\geq0}u^n$ converge absolument, donc converge.\\
    Pour $n\in\N, ~ \|u^n\|\leq \|u\|^n$ puis par comparaison (positifs) avec une série géométrique, $\sum\|u^n\|$ converge.\\
    Pour $n\in\N, ~ (e-u)\sum_{n=0}^nu^k=e-u^{n+1}$ et $\|u^{n+1}\|\leq \|u\|^{n+1}\to0$ donc $u^{n+1}\to0$ donc $(e-u)\sum_{n=0}^{\infty}u^n\to e$ par passage à la limite donc $(e-u)$ est inversible et $(e-u)^{-1}=\sum_{n=0}^{\infty}u^n$.\\
    \boxed{2.} Soit $A\in\GL_n(\R)$, $\exists R > 0, ~ B(A,R)\subset\GL_n(\R)$. Soit $H\in B(0,R)$.\\
    On a $f(A+H)-f(A)=(A+H)^{-1} - A^{-1} = (A(I_n + A^{-1}H))^{-1} - A^{-1} = (I_n + A^{-1}H)^{-1}A^{-1} - A^{-1}$.\\
    Posons $u=A^{-1}H$, alors par le 1, $\|u\|<1 \ra I_n + u\in\GL_n(\R) \ra (I_n + u)^{-1}=\sum_{n=0}^{\infty}(-1)^nu^n$.\\
    Donc avec $H\in B(A,R)$ et $\|H\|<\frac{1}{\|A^{-1}\|}$ : $\|u\|=\|A^{-1}H\| < 1$ puis $(I_n + u)^{-1}=(I_n + A^{-1}H)^{-1}=\sum_{n=0}^{+\infty}(-1)^n(A^{-1}H)^n$.\\
    Donc $f(A+H)-f(A)=\sum_{n=0}^{\infty}(-1)^n(A^{-1}H)^nA^{-1}=-A^{-1}HA^{-1}+\sum_{n=2}^{+\infty}(-1)^n(A^{-1}H)^nA^{-1}$.\\
    Donc $f(A+H)-f(A)=L(H)+\|H\|\e(H)$, avec $L(H)=-A^{-1}HA^{-1}$; $L$ linéaire et $\e(H)\to0$.\\
    Alors $df_A(H)=-A^{-1}HA^{-1}$.
\end{exercice}

\begin{exercice}{}{}
    \begin{enumerate}
        \item Montrer que $A\mapsto\exp(A)$ est continue sur $\M_n(\R)$.
        \item Calculer la différentielle de $A\mapsto \exp(A)$ en 0.
    \end{enumerate}
    \tcblower
    \boxed{1.}
    \begin{equation*}
        \forall A \in \M_n(\C), ~ \exp(A) = \sum_{n=0}^{+\infty}\frac{A^n}{n!}
    \end{equation*}
    Théorème de continuité des séries de fonctions :\\
    $\hra$ $f_n:A\mapsto \frac{A^n}{n!}$ est continue pour tout $n\in\N$.\\
    $\hra$ $\forall R \geq 0, ~ \forall A \in B'(0,R), ~ \left\|\frac{A^n}{n!}\right\|\leq \frac{\|A\|^n}{n!} \leq \frac{R^n}{n!}$ donc $\|f_n\|_{\infty}^{B'(0,R)}\leq\frac{R^n}{n!}$.\\
    Donc $\sum_{n\geq0}u_n$ $\CN$ puis $\CU$ sur $B'(0,R)$, il y a convergence uniforme sur tout compact de $\M_n(\C)$.\\
    Par théorème, $\exp$ est continue sur $\M_n(\C)$.\\
    \boxed{2.} On cherche un développement limité à l'ordre 1 en 0 de l'exponentielle.\\
    $\hra$ $\exp(0+H)=\exp(0) + L(H) + \|H\|\e(H)$ avec $L\in\L(\M_n(\C))$ et $\e(H)\xrightarrow[H\to0]{}0$.\\
    Soit $H\in\M_n(\C)$. 
    \begin{equation*}
        \exp(H) = \sum_{n=0}^\infty\frac{H^n}{n!} = I_n + H + \sum_{n=2}^{\infty} \frac{H^n}{n!}
    \end{equation*}
    Par IT et passage à la limite : $\left\|\sum_{n=2}^\infty\frac{H^n}{n!}\right\|\leq\sum_{n=2}^{\infty}\frac{\|H^n\|}{n!}\leq\sum_{n=2}^\infty\frac{\|H\|^n}{n!}=e^{\|H\|}-1-\|H\| \sim_0 \frac{\|H\|^2}{2}$.\\
    Donc par majoration, $\sum_{n=2}^\infty\frac{H^n}{n!}=o(\|H\|)$, finalement : $\exp$ est différentiable en 0 et $d\exp_0=\Id$.
\end{exercice}

\begin{thm}{}{}
    Soient $f,g:U\subset E \to F$, $a\in U$ et $\l\in\R$. 
    \begin{enumerate}
        \item Si $f$ et $g$ sont différentiables en $a$, alors $f+\l g$ aussi et $d(f+\l g)_a =df_a + \l dg_a$.
        \item Si $f$ et $g$ sont différentiables sur $U$, alors $f+\l g$ aussi et $d(f+\l g) = df + \l dg$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Par hypothèse, $f$ et $g$ admettent des DL1 en $a$ :
    \begin{equation*}
        f(a+h) = f(a) + L_f(h) + \|h\|\e_f(h) \et g(a+h) = g(a) + L_g(h) + \|h\|\e_g(h).
    \end{equation*}
    Où $L_f,L_g\in\L(E,F)$ et $\e_f(h),\e_g(h)\xrightarrow[h\to0]{}0$. Puis :
    \begin{equation*}
        (f + \l g)(a+h) = f(a+h) + \l g(a+h) = (f + \l g)(a) + (L_f + \l L_g)(h) + \|h\|(\e_f + \l \e_g)(h)
    \end{equation*} 
    Où $L_f+\l L_g \in \L(E,F)$ et $\e_f + \l \e_g \xrightarrow[h\to0]{}0$, donc $(f+\l g)$ est différentiable en $a$ et $d(f+\l g)_a=df_a+\l dg_a$.
\end{thm}

\begin{prop}{}{}
    Soient $f:U\subset E \to F$, $\l:U\to\R$ et $a\in U$.
    \begin{enumerate}
        \item Si $f$ et $\l$ sont différentiables en $a$, alors $\l f$ aussi et $d(\l f)_a = \l(a)d f_a + d\l_af(a)$.
        \item Si $f$ et $\l$ sont différentiables sur $U$, alors $\l f$ aussi et $d(\l f) = \l d_f + d\l f$.
    \end{enumerate}
\end{prop}

\begin{prop}{}{}
    Soient $f:U\subset E \to F$, $g:V\subset F \to G$ et $a\in U$, $b=f(a)\in V$.
    \begin{enumerate}
        \item Si $f$ est différentiable en $a$ et $g$ différentiable en $b$, alors $g\circ f$ est différentiable en $a$ et $d(g\circ f)_a = dg_{f(a)}\circ df_a$.
        \item Si $f$ est différentiable sur $U$ et $g$ différentiable sur $V$, avec $f(U)\subset V$, alors $(g\circ f)$ est différentiable sur $U$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Supposons $f$ différentiable en $a$ et $g$ différentiable en $b$, alors :
    \begin{equation*}
        f(a+h) = f(a) + L_f(h) + \|h\|\e_f(h) \et g(b+k) = g(a) + L_g(k) + \|k\|\e_g(k).
    \end{equation*}
    Avec $L_f\in\L(E,F)$, $L_g\in\L(F,G)$, $\e_f(h)\xrightarrow[h\to 0]{}0$ et $\e_g(k)\xrightarrow[k\to0]{}0$.
    \begin{align*}
        (g\circ f)(a+h) &= g(f(a+h)) = g(f(a) + L_f(h) + \|h\|\e_1(h)) = g(f(a)) + L_g(k) + \|k\|\e_g(k)\\
        &=(g\circ f)(a) + L_g\circ L_1(h) + \|h\|L_g\circ \e_f(h) + \|k\|\e_g(k).
    \end{align*}
    Avec $k=\L_f(h) + \|h\|\e_f(h)$.\\
    $\hra$ $L_g\circ L_f \in \L(E,G)$;\\
    $\hra$ $\e_f(h)\to0$ et $L_g$ continue sur $F$ donc en 0 : $L_g(\e_f(h))\to L_g(0)=0$ donc $\|h\|L_g\circ \e_f(h)=_0o(\|h\|)$.\\
    $\hra$ $k\xrightarrow[h\to 0]{}0$ par continuité de $L_f$ en 0. Donc $\e_g(k)\xrightarrow[h\to0]{}0$ par composition des limites.\\
    Puis $L_f$ est linéaire sur $E$ evdf donc : $\exists C\geq0, ~ \forall h \in E, ~ \|L_f(h)\|_F\leq C\|h\|_E$.\\
    Puis $\|h\|_E\leq \|L_f(h)\|_F + \|h\|_E\|\e_f(h)\|_F \leq \|h\|(C+\|\e_f(h)\|_F)$ puis $\|k\|_F\e_g(k)=o(\|h\|)$.\\
    Finalement, $(g\circ f)(a+h)=(g\circ f)(a) + L_g\circ L_f(h) + \|h\|\e(h)$ donc $d(g\circ f) = dg_{f(a)}\circ df_a$.
\end{prop}

\bf{Remarque.} Si $E=F=\R$ : $f'(a)=df_a(1)$ et $g'(b)=dg_b(1)$.\\
Alors $(g\circ f)'(a)=d(g\circ f)_a(1)=(dg_{f(a)}\circ df_a)(1)=dg_{f(a)}(f'(a))=f'(a)dg_{f(a)}(1)=f'(a)g'(f(a))=f'(a)(g'\circ f)(a)$.

\begin{defi}{Matrice jacobienne.}{}
    Soit $f:U\subset E\to F$ et $a\in U$ tel que $f$ est différentiable en $a$.\\
    Soit $\B_1=(e_1,...,e_p)$ base de $E$ et $\B_2=(e_1',...,e'_n)$ base de $F$. La matrice jacobienne de $f$ en $a$ est définie par :
    \begin{equation*}
        \Jac_a f = \Mat_{\B_2,\B_1}(df_a) \in \M_{n,p}(\R)
    \end{equation*}
    \tcblower
    On sait que $\forall v \in E$ avec $v$ non nul, $df_a(v) = Df_v(a)$ et $df_a(e_i)=Df_{e_i}(a)=\frac{\6 f}{\6 x_i}(a)$.\\
    Posons $f:x\mapsto \sum_{j=1}^nf_j(x)e'_j$ avec $f_1,...,f_n$ sont les applications coordonnées de $f$ dans la base $\B_2$ de $F$.\\
    Finalement : $\Jac_a f = \left( \frac{\6 f_i}{\6 x_j}(a) \right)_{i,j}$
\end{defi}

\begin{ex}{}{}
    Soit $f(x,y)=((\sin x)e^y,x^2y^3,x+\cos y)$ pour tout $(x,y)\in\R^2$.\\
    Déterminer la matrice jacobienne de $f$ sur $\R^2$.
    \tcblower
    Notons $f_1(x,y)=(\sin x)e^y$ ; $f_2(x,y)=x^2y^3$ ; $f_3(x,y)=x+\cos y$.
    \begin{equation*}
        \Jac_af=\begin{pmatrix} (\cos x)e^y & (\sin x)e^y \\ 2xy^2 & 3x^2y^2 \\ 1 & -\sin y\end{pmatrix}
    \end{equation*}
\end{ex}

\bf{Traduction matricielle du DL1.}

Supposons que $f$ est différentiable en $a$ : $f(a+h) = f(a) + df_a(h) + \|h\|\e(h)$ avec $\e(h)\to0$.\\
Deux vecteurs sont égaux ssi leurs coordonnées dans une base sont égaux. Avec les notations précédentes :
\begin{equation*}
    \begin{pmatrix}
        f_1(a+h) \\ \vdots \\ f_n(a+h)
    \end{pmatrix} = \begin{pmatrix}
        f_1(a) \\ \vdots \\ f_n(a)
    \end{pmatrix} + \begin{pmatrix}
        \frac{\6 f_1}{\6 x_1}(a) & \cdots & \frac{\6 f_1}{\6 x_p}(a) \\ \vdots & \ddots & \vdots \\ \frac{\6 f_n}{\6 x_1}(a) & \cdots & \frac{\6 f_n}{\6 x_p}(a)
    \end{pmatrix} \begin{pmatrix}
        h_1 \\ \vdots \\ h_p
    \end{pmatrix} + \|h\|\begin{pmatrix}
        \e_1(h) \\ \vdots \\ \e_n(h)
    \end{pmatrix}
\end{equation*}
i.e. $\forall j \in \lb1,n\rb, ~ f_j(a+h)=f_j(a) + \sum_{i=1}^p h_i\frac{\6 f_j}{\6_xi}(a) + \|h\|\e_j(h)$.

\bf{Traduction matricielle des opérations.}

Supposons $f$ et $g$ différentiables en $a$ et $\l\in\R$, $d(f+\l g)_a = df_a + \l dg_a$ i.e. $\Jac_a(f+\l g)=\Jac_a f + \l \Jac_a g$.\\
Supposons $f$ et $\l$ différentiable en $a$, $d(\l f)_a = \l(a)df_a+d\l_af(a)$ i.e. $\Jac_a(\l f) = \l(a)\Jac_a(f)+f(a)\Jac_a(\l)$.\\
Supposons $f$ différentiable en $a$ et $g$ en $f(a)$, $d(g\circ f)_a = dg_{f(a)}\circ df_a$ i.e. $\Jac_a(g\circ f)=\Jac_{f(a)}(g)\times \Jac_a(f)$.

\begin{defi}{}{}
    Soit $f:U\subset E \to \R$ avec $(E,\Lg\.,\.\Rg)$ euclidien et $a\in U$. Si $f$ est différentiable en $a$, alors 
    \begin{equation*}
        \exists ! v \in E, ~ \forall h \in E, ~ df_a(h) = \Lg v, h \Rg.
    \end{equation*}
    Ce vecteur $v$ est appelé vecteur gradient de $f$ en $a$ et noté $\grad f_a$ ou $\nabla f_a$.
\end{defi}

\begin{prop}{Coordonnées du vecteur gradient dans $\B$ b.o.n.}{}
    Soit $(E,\Lg\.,\.\Rg)$ euclidien et $\B=(e_1,...,e_n)$ une b.o.n. de $E$. Soit $f:U\subset E \to \R$ et $a\in U$.\\
    Si $f$ est différentiable en $a$, alors : $\ds\nabla f_a = \sum_{i=1}^n \frac{\6 f}{\6 x_i}(a)e_i$.
    \tcblower
    Rappel : $\forall x \in E, ~ x = \sum_{i=1}^n \Lg x, e_i \Rg e_i$ (car b.o.n.).\\
    Ici, $\nabla f_a = \sum_{i=1}^n\Lg \nabla f_a, e_i\Rg e_i = \sum_{i=1}^n df_a(e_i)e_i = \sum_{i=1}^n Df_{e_i}(a)e_i = \sum_{i=1}^n\frac{\6 f}{\6 x_i}(a)e_i$.
\end{prop}

On réécrit le DL1 de $f$ en $a$ : $f(a+h) = f(a) + \Lg \nabla f_a, h\Rg + o(\|h\|)$ avec $\Lg \nabla d_a, h) = \sum_{i=1}^n \frac{\6 f}{\6 x_i}(a)h_i$.\\
Par Cauchy-Schwarz : $|\Lg \nabla f_a, v \Rg| \leq \|\nabla f\|\|v\|$ donc $|Df_v(a)|\leq\|\nabla f_a\|$ pour tout $v$ unitaire.\\
Avec égalité ssi $v$ est positivement lié à $\nabla f_a$, donc $\nabla f_a\neq0$ indique la direction de plus forte pente.\\
Si $\nabla f_a=0$, alors $a$ est un point critique, donc $f$ peut admettre en $a$ un extremum local.

\end{prop}

\begin{prop}{Règle de la chaîne.}{}
    Soit $f:U\subset\R^p\to\R^n$ et $g:V\subset\R^n\to\R$ différentiables.

        \forall i \in \lb1,p\rb, ~ \frac{\6(g\circ f)}{\6 x_i} = \sum_{j=1}^n \frac{\6 g}{\6 y_j}(f(x))\frac{\6 f_j}{\6 x_i}(x).

    \tcblower

\end{prop}

\begin{exercice}{Calcul du gradient en coordonnées polaires.}{}
    Soit $f:(x,y)\mapsto f(x,y)$, on pose $x=r\cos\theta$ et $y=r\sin\theta$ pour $(x,y)\neq(0,0)$, $r>0$ et $\theta\in[0,2\pi[$.\\
    Alors $f(x,y)=f(r\cos\theta, r\sin\theta)=F(r,\theta)$ avec $F=f\circ g$ où $g(r,\theta)=(r\cos\theta,r\sin\theta)$.
    \begin{align*}
        &\frac{\6 F}{\6 r}(r,\theta) = \frac{\6 f}{\6 x}(x,y)\frac{\6 g_1}{\6 r}(r,\theta) + \frac{\6 f}{\6 y}(x,y)\frac{\6 g_2}{\6 r}(r,\theta) = \cos\theta\frac{\6 f}{\6 x}(x,y) + \sin\theta\frac{\6 f}{\6 y}(x,y).\\
        &\frac{\6 F}{\6 \theta}(r,\theta) = \frac{\6 f}{\6 x}(x,y)\frac{\6 g_1}{\6 \theta}(r, \theta) + \frac{\6 f}{\6 y}(x,y)\frac{\6 g_2}{\6 \theta}(r,\theta) = -r\sin\theta\frac{\6 f}{\6 x}(x,y)+r\cos\theta\frac{\6 f}{\6 y}(x,y).
    \end{align*} 
    Puis $\nabla f(x,y) = \left(\frac{\6 f}{\6 x}(x,y), \frac{\6 f}{\6 y}(x,y) \right)$, coordonnées du gradient dans la base canonique de $\R^2$ (b.o.n.).\\
    Notons $\v{i}=(1,0)$ et $\v{j}=0,1$, alors :
    \begin{equation*}
        \nabla f(x,y) = \frac{\6 f}{\6 x}(x,y)\v{i} + \frac{\6 f}{\6 y}(x,y)\v{j}.
    \end{equation*}
    Or d'après les calculs précédents :
    \begin{equation*}
        \begin{pmatrix} \frac{\6 f}{\6 x}(x,y) & \frac{\6 f}{\6 y}(x,y) \end{pmatrix} = \begin{pmatrix}\frac{\6 F}{\6 r}(r,\theta) & \frac{\6 F}{\6 \theta}(r,\theta)\end{pmatrix}\begin{pmatrix}\cos \theta & -r\sin \theta \\ \sin\theta & r\cos\theta\end{pmatrix}^{-1} = \frac{1}{r}\begin{pmatrix}\frac{\6 F}{\6 r}(r,\theta) & \frac{\6 F}{\6 \theta}(r,\theta)\end{pmatrix}\begin{pmatrix}r\cos\theta&r\sin\theta\\-\sin\theta&\cos\theta\end{pmatrix}
    \end{equation*}
    Donc $\nabla f(x,y) = \left( \cos\theta \frac{\6 F}{\6 r}(r,\theta) - \frac{\sin \theta}{r}\frac{\6 F}{\6 \theta}(r,\theta) \right)\v{i} + \left( \sin\theta\frac{\6 F}{\6 r}(r,\theta) + \frac{\cos\theta}{r}\frac{\6 F}{\6 \theta}(r,\theta) \right)\v{j}$.\\
    Puis $\nabla f(x,y) = \frac{\6 F}{\6 r}(r,\theta)(\cos \theta \v{i} + \sin\theta\v{j}) + \frac{1}{r}\frac{\6 F}{\6 \theta}(r,\theta)(-\sin\theta\v{i} + \cos\theta\v{j})$.
\end{exercice}

\section{Classe \texorpdfstring{$\cursive{C}^k$}{Lg}.}

\begin{defi}{Classe $\cursive{C}^1$.}{}
    Soit $f:U\subset E \to F$ avec $E,F$ deux $\R$-evdf et $U$ un ouvert de $E$.\\
    On dit que $f$ est de classe $\cursive{C}^1$ sur $U$ ssi $f$ est différentiable sur $U$ et si $df$ est continue sur $U$.
\end{defi}

\begin{thm}{Théorème fondamental (Admis).}{}
    Soit $f:U\subset E\to F$ avec $E,F$ deux $\R$-evdf et $U$ un ouvert de $E$.
    \begin{enumerate}
        \item Si $f$ est $\cursive{C}^1$ sur $U$, alors $\forall v \in E, ~ v\neq0 \ra D_vf$ est continue sur $U$.
        \item \fbox{Si $f$ admet $n$ dérivées partielles continues sur $U$, alors $f$ est $\cursive{C}^1$ sur $U$.}
    \end{enumerate}
\end{thm}

\begin{thm}{}{}
    La classe $\cursive{C}^1$ est stable par combinaisons linéaires, par produit et par composition.
\end{thm}

\begin{corr}{}{}
    Les fonctions polynomiales de plusieurs variables sont de classe $\cursive{C}^1$ sur leur domaine de définition.
\end{corr}

\begin{prop}{Dérivée le long d'un arc.}{}
    Soit $\g:I\subset\R\to U$ $\cursive{C}^1$ sur $I$ et $f:U\to\R$ $\cursive{C}^1$ sur $U$. Alors $f\circ \g$ est $\cursive{C}^1$ sur $I$ et
    \begin{equation*}
        \forall t \in I, ~ (f\circ \g)'(t) = \Lg \g'(t), \nabla f(\g(t)) \Rg = df_{\g(t)}(\g'(t)).
    \end{equation*}
    \tcblower
    On sait déjà que $f\circ \g$ est $\cursive{C}^1$ sur $I$ par composition, et $d(f\circ \g)(t)=df_{\g(t)}\circ d_{\g}(t)$.\\
    Ici, $\forall t \in \R, ~ (f\circ \g)'(t) = d(f\circ \g)_t(1) = (df_{\g(t)}\circ d\g_t)(1)=df_{\g(t)}(d\g_t(1))=df_{\g(t)}(\g'(t))$.
\end{prop}

\begin{prop}{Théorème fondamental de l'analyse ?}{}
    Soit $f:U\subset E\to\R$ $\cursive{C}^1$ sur $U$ et $(a,b)\in U^2$. Soit $\g:[0,1]\to U$ $\cursive{C}^1$ telle que $\g(0)=a$ et $\g(1)=b$. Alors :
    \begin{equation*}
        f(b) - f(a) = \int_0^1 df_{\g(t)}(\g'(t))\dt
    \end{equation*}
    \tcblower
    On a $f(b)-f(a)=f(\g(1))-f(\g(0))=f\circ\g(1) - f\circ\g(0)$, et $f\circ\g$ est $\cursive{C}^1$ sur $[0,1]$ à valeurs réelles. Par TFA :
    \begin{equation*}
        (f\circ \g)(1) - (f\circ\g)(0) = \int_0^1 (f\circ \g)'(t)\dt = \int_0^1 df_{\g(t)}(\g'(t))\dt
    \end{equation*}
\end{prop}

\bf{Remarque.} Dans $\R$, avec $f$ de classe $\cursive{C}^1$ au sens classique :
\begin{equation*}
    f(b) - f(a) = \int_a^b f'(t)\dt = \int_0^1 f'(u a + (1-u)b)(a-b)\du
\end{equation*}

\begin{prop}{Caractérisation des fonctions constantes sur $U$.}{}
    Soit $f:U\subset E \to \R$ de classe $\cursive{C}^1$ sur $U$ connexe par arcs. Soit $\B=(e_1,...,e_n)$ une base de $E$.\\ 
    Alors $f$ est constante sur $U$ ssi toutes ses dérivées partielles sont nulles sur $U$.
    \tcblower
    \boxed{\la} Supposons que toutes les dérivées partielles de $f$ sont nulles sur $U$. Soit $(a,b)\in U$.\\
    Alors $\forall t \in [0,1]$, on pose $\g(t)=tb+(1-t)a$, de classe $\cursive{C}^1$ sur $[0,1]$. Alors par dérivée le long d'un arc :
    \begin{equation*}
        f(b) - f(a) = \int_0^1 df_{\g(t)}(\g'(t))\dt = \int_0^1 \sum_{i=1}^n \frac{\6 f}{\6 x_i}(\g(t))\g_i'(t)\dt = 0
    \end{equation*}
    Avec $\g=(\g_1,...,\g_n)$, on a donc $f(a)=f(b)$, donc $f$ est bien constante sur $U$.\\
    \boxed{\ra} Si $f$ est constante sur $U$, alors $\forall a \in E, ~ \forall v \in E, ~ v\neq 0 \ra Df_v(a)=0$.\\
    En particulier, les dérivées partielles de $f$ sont nulles sur $U$.
\end{prop}

Dans le cas général $U$ connexe par arcs, on sait qu'il existe un chemin continu à valeurs dans $U$ qui joint $a$ et $b$.

\begin{exercice}{}{}
    Soit $f:\R^3\to\R$ de classe $\cursive{C}^1$ sur $\R^3$ et $a\in\R$.\\
    On dit que $f$ est homogène d'ordre $a$ ssi $\forall (x,y,z)\in\R^3, ~ \forall t > 0, ~ f(tx,ty,tz)=t^af(x,y,z)$.\\
    Montrer que $f$ est homogène d'ordre $a$ ssi $\forall (x,y,z)\in\R^3, ~ x\frac{\6 f}{\6 x}(x,y,z)+y\frac{\6 f}{\6 y}(x,y,z)+z\frac{\6 f}{\6 z}(x,y,z)=af(x,y,z)$.
    \tcblower
    \boxed{\ra} Par hypothèse, $\forall t > 0, ~ \forall (x,y,z)\in\R^3, ~ f(tx,ty,tz)=t^af(x,y,z)$. Soit $(x,y,z)\in\R^3$.
    \begin{equation*}
        g(t) := f(tx,ty,tz) = (f\circ\phi)(t) \nt{ avec } \phi(t)=(tx,ty,tz).
    \end{equation*}
    On dérive par rapport à $t$ en utilisant la règle de la chaîne.
    \begin{equation*}
        g'(t) = x \frac{\6 f}{\6 x}(tx,ty,tz) + y\frac{\6 f}{\6 y}(tx,ty,tz) + z\frac{\6 f}{\6 z}(tx,ty,tz).
    \end{equation*}
    D'autre part, $g'(t)=at^{a-1}f(x,y,z)$, en évaluant les deux expressions en $t=1$ :
    \begin{equation*}
        x\frac{\6 f}{\6 x}(x,y,z) + y\frac{\6 f}{\6 y}(x,y,z) + z\frac{\6 f}{\6 z}(x,y,z) = af(x,y,z).
    \end{equation*}
    \boxed{\la} Posons $g(t)=f(tx,ty,tz)$ pour $t>0$ et $(x,y,z)$ fixés.\\
    Cherchons une expression de $g$ via une équation différentielle.
    \begin{align*}
        g'(t) &= x\frac{\6 f}{\6 x}(tx, ty,tz) + y\frac{\6 f}{\6 y}(tx,ty,tz) + z\frac{\6 f}{\6 z}(tx,ty,tz)\\
        &=\frac{1}{t}\left( tx\frac{\6 f}{\6 x}(tx,ty,tz) + ty \frac{\6 f}{\6 y}(tx,ty,tz) + tz\frac{\6 f}{\6 z}(tx,ty,tz) \right)\\
        &= \frac{1}{t}af(tx,ty,tz) = \frac{a}{t}g(t).
    \end{align*}
    Donc $g$ est solution sur $\R_+^*$ de $y'=\frac{a}{t}y$ donc $\exists \l \in \R, ~ \forall t > 0, ~ g(t)=\l e^{a\ln t}=\l t^a$.\\
    Puis $\l=g(1)=f(x,y,z)$.
\end{exercice}

\begin{defi}{Vecteurs tangents à une partie.}{}
    Soit $X\subset E$, $v\in E$ et $x\in X$ avec $E$ un $\R$-evdf.\\
    On dit que $v$ est tangent à la partie $X$ au point $x$ ssi $\exists \e>0, ~ \exists\phi:~]-\e,\e[\to X, ~ \phi(0)=x$ et $\phi'(0)=v$.
\end{defi}

\begin{exercice}{}{}
    Pour $f:I\subset\R\to\R$, on note $X=\{(x,f(x)); ~ x\in I\}$ le graphe de $f$.\\
    Déterminer les vecteurs tangents à la partie $X$ en $M_0 = (x_0,f(x_0))\in X$ et en déduire une équation de la tangente à $f$ en $x_0$.
\end{exercice}
