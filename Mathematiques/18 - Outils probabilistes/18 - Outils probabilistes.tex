\begin{defi}{EspÃĐrance.}{}

\bf{Remarque.} $(x\Pr(X=x))_{x\in X(\O)}$ est sommable ssi $(|x|\Pr(X=x))_{x\in X(\O)}$ est sommable ssi $\sum_{x\in X(\O)}|x|\Pr(X=x)<\infty$.\\
\bf{Remarque.} On rappelle que si $\sum_{n\geq 0} u_n$ $\CA$, alors $\sum u_n$ est commutativement convergente. 

\begin{ex}{}{}
    $\bullet$ Si $X$ suit une loi uniforme sur $\{x_1,...,x_n\}$ avec $n\in\N^*$, alors $\E(X)=\sum_{i=1}^n\frac{x_i}{n}=\frac{1}{n}(x_1+...+x_n)$.\\
    $\bullet$ Si $X\sim\B(p)$, alors $\E(X)=0\cdot\Pr(X=0)+1\cdot\Pr(X=1)=p$.\\
    $\bullet$ Si $X\sim\B(n,p)$, alors $\E(X)=np$.
\end{ex}

\begin{thm}{ThÃĐorÃĻme de transfert.}{}
    \begin{enumerate}
        \item Soient $X,Y$ v.a.d. sur $(\O,\A,\Pr)$ et $f:X(\O)\times Y(\O)\to\R$.
        \begin{equation*}
            \E(f(X,Y)) = \sum_{\substack{x\in X(\O)\\y\in Y(\O)}}f(x,y)\Pr(X=x \cap Y=y).
        \end{equation*}
        \item Soit $X$ v.a.d. sur $(\O,\A,\Pr)$ et $f:X(\O)\to \R$.
        \begin{equation*}
            \E(f(X))=\sum_{x\in X(\O)}f(x)\Pr(X=x).
        \end{equation*}
    \end{enumerate}
    \tcblower
    \boxed{1.} Posons $Z=f(X,Y)$ v.a.d. sur $(\O,\A,\Pr)$.\\
    On a $Z\in\L^1$ ssi $(|z|\Pr(Z=z))_{z\in Z(\O)}$ est sommable.
    \begin{equation*}
        \forall z \in Z(\O), ~ (Z=z)=(f(X,Y)=z)=\bigcup_{f(x,y)=z}(X=x\cap Y=y) 
    \end{equation*}
    \begin{equation*}
        \Pr(Z=z)=\sum_{f(x,y)=z}\Pr(X=x\cap Y=y) = \sum \1_{f(x,y)=z}\Pr(X=x\cap Y=y).
    \end{equation*}
    puis $|z|\Pr(Z=z)=\sum|z|\1_{f(x,y)=z}\Pr(X=x\cap Y=y)$
    \begin{equation*}
        \E(Z) = \sum_{z\in\R}\left( \sum_{x,y} z\1_{f(x,y)=z} \Pr(X=x\cap Y=y)\right) = \sum_{x,y}f(x,y)\Pr(X=x\cap Y=y).
    \end{equation*}
    \boxed{2.} Posons $g(x,y)=f(x)$ pour tout $x\in X(\O)$ et $y\in Y(\O)$.
    \begin{equation*}
        \E(f(X)) = \E(g(X,Y)) = \sum_{x,y} g(x,y)\Pr(X=x\cap Y=y) = \sum_{x\in X(\O)}f(x) \left( \sum_{y\in Y(\O)} \Pr(X=x\cap Y=y)\right)
    \end{equation*}
    avec $\sum_y \Pr(X=x\cap Y=y) = \Pr(X=x)$ (s.c.e)
\end{thm}

\begin{prop}{Domination.}{}
    Soit $X,Y$ deux v.a.d. sur $(\O,\A,\Pr)$ telles que $|X|\leq Y$ et $Y\in\L^1$, alors $X\in\L^1$.
    \tcblower
    VÃĐrifions que $\forall x \in X(\O), ~ \forall y \in Y(\O), |x|\Pr(X=x\cap Y=y)\leq y\Pr(X=x\cap Y=y)$.\\
    Soit $x\in X(\O)$ et $y\in Y(\O)$.\\
    $\bullet$ Si $|x|\leq y$, alors $|x|\Pr(X=x\cap Y=y)\leq y\Pr(X=x\cap Y=y)$.\\
    $\bullet$ Sinon, $|x|>y$ alors $(X=x\cap Y=y)=\0$ par hyp. puis $\Pr(X=x\cap Y=y)=0$ donc l'inÃĐgalitÃĐ est vraie.\\
    Supposons $Y\in\L^1$. Alors $(y\Pr(X=x\cap Y=y))$ est sommable donc par domination, $|x|\Pr(X=x\cap Y=y)$ est sommable.
    \begin{equation*}

    \end{equation*}
\end{prop}

\begin{prop}{EspÃĐrance et Variance des lois usuelles.}{}
    \begin{enumerate}
        \item Soit $X\sim\B(p)$, alors $\E(X)=p$ et $V(X)=p(1-p)$.
        \item Soit $X\sim\B(n,p)$, alors $\E(X)=np$ et $V(X)=np(1-p)$.
        \item Soit $X\sim\cursive{P}(\l)$, alors $\E(X)=V(X)=\l$.
        \item Soit $X\sim\cursive{G}(p)$, alors $\E(X)=\frac{1}{p}$ et $\V(X)=\frac{1-p}{p^2}$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Soit $X\sim \B(p)$. $\E(X)=\Pr(X=1)=p$ et $X^2=X$ donc $V(X)=\E(X^2)-\E(X)^2 = p-p^2 = p(1-p)$.\\
    \boxed{2.} Soit $X_1,...,X_n\sim\B(p)$, alors $X:=X_1+...+X_n\sim\B(n,p)$; $\E(X)=\E(X_1+...+X_n)=\E(X_1)+...+\E(X_n)=np$.\\
    Puis $V(X)=V(X_1)+...+\V(X_n)=np(1-p)$.\\
    \boxed{3.} Soit $X\sim\cursive{P}(\l)$. $\E(X)=\l$ $(\ref{prop:esp})$, et $\E(X(X-1))=\sum_{n=0}^\infty n(n-1)\Pr(X=n)=\l^2e^{-\l}e^\l=\l^2$.\\
    Puis $\E(X^2)=\E(X(X-1)+X)=\E(X(X-1))+\E(X)=\l^2+\l$ puis $V(X)=\E(X^2)-\E(X)^2=(\l^2+\l)-\l^2 = \l$.\\
    \boxed{4.} Soit $x\sim\cursive{G}(p)$. $\E(X)=\frac{1}{p}$ $(\ref{prop:esp})$ puis $\E(X(X-1))=\sum_{n=1}^{+\infty}n(n-1)\Pr(X=n)=\frac{2p(1-p)}{(1-(1-p))^3}=\frac{2(1-p)}{p^2}$.\\
    Puis $\E(X^2)=\E(X(X-1)+X)=\frac{2(1-p)}{p^2} + \frac{1}{p}$ puis $V(X)=\E(X^2)-\E(X)^2 = \frac{2(1-p)}{p^2} + \frac{p-1}{p^2} = \frac{1-p}{p^2}$.
\end{prop}

\section{Covariance.}

\begin{defi}{}{}
    Soient $X,Y\in\L^2$. On dÃĐfinit la covariance de $X,Y$ : $\cov(X,Y)=\E\left( \left( X-\E(X) \right) \left( Y - \E(Y) \right)\right)$.
\end{defi}

\vspace*{-0.2cm}

\begin{prop}{}{}
    Si $X,Y$ sont deux v.a.d. sur $(\O,\A,\Pr)$. alors $\cov(X,Y)=\E(XY)-\E(X)\E(Y)$
\end{prop}

\vspace*{-0.2cm}

\bf{Remarque.} $X\ind Y \ra \cov(X,Y) = 0$, rÃĐciproque fausse.

\begin{ex}{}{}
    Soit $a,b\in\R^2$, $a<b$ et $X$ sur $(\O,\A,\Pr)$ suivant la loi uniforme.\\
    Alors $X^2$ suit la loi uniforme, et $X,X^2$ admettent des moments de tout ordre.\\
    Puis $\Pr(X=a\cap X^2=b^2)\neq \Pr(X=a)\Pr(X^2=b^2)$, donc $X\not\ind X^2$.\\
    Puis $\cov(X,X^2)=\E(X^3) - \E(X)\E(X^2)$ or $\E(X^3)=0$ (transfert) et $\E(X=0)$, donc $\cov(X,X^3)=0$.
\end{ex}

\vspace*{-0.2cm}

\begin{prop}{}{}
    Soit $X,Y$ deux v.a.d. sur $(\O,\A,\Pr)$ dans $\L^2$, alors $(X,Y)\mapsto \cov(X,Y)$ est symÃĐtrique, bilinÃĐaire et positive.
    \tcblower
    $\cov(X,Y)=\E(XY)-\E(X)\E(Y)=\cov(Y,X)$; $\cov(X+\l X',Y)=\E((X+\l X'), Y) - \E(X+\l X')\E(Y)=\cov(X,Y)+\l\cov(X',Y)$.\\
    LinÃĐaire Ã  gauche et sym. donc bilinÃĐaire. $\cov(X,X)=V(X)\geq0$ et $\cov(X,X)=0\ra V(X)=0\ra X=\cste$ presque surement.
\end{prop}

\begin{prop}{}{}
    Soient $X,Y$ deux v.a.d. sur $(\O,\A,\Pr)$ dans $\L^2$ et $\l\in\R$, alors $V(X+\l Y)=V(X)+2\l\cov(X,Y)+\l^2V(Y)$.\\

    \begin{equation*}
        V(\sum_{i=1}^n \l_i X_i) = \sum_{i=1}^n \l_i\l_j\cov(X_i,X_j) = \sum_{i=1}^n\l_i^2 V(X_i) + 2\sum_{i\neq j}\l_i\l_j\cov(X_i, X_j).

\end{prop}

        \E\left( \frac{S_n}{n} \right) = \frac{1}{n}\sum_{i=1}^n \E(X_i) = \E(X_1) \et V\left( \frac{S_n}{n} \right) = \frac{1}{n^2}V(S_n) = \frac{1}{n^2}\sum_{i=1}^nV(X_i)=\frac{V(X_1)}{n}.
    \end{equation*}
    D'aprÃĻs BienaymÃĐ-Tchebychev : 
    \begin{equation*}
        \Pr\left( \left|\frac{S_n}{n} - m \right| \geq \e \right) \leq \frac{V(\frac{S_n}{n})}{\e^2} = \frac{1}{n}\.\frac{V(X_1)}{\e^2} \xrightarrow[n\to+\infty]{}0
    \end{equation*}
\end{corr}
