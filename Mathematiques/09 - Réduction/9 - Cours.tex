\section{PolynûÇmes d'endomorphismes et de matrices.}

\begin{defi}{}{}
    Soit $E$ un $\K$-ev, $P=\sum a_nX^n \in \K[X]$ et $u\in\L(E)$.
    \begin{equation*}
        P(u) = a_0 \Id_E + a_1u + ... + a_nu^n.
    \end{equation*}
    \bf{Remarque.} $\forall P \in \K[X], ~ \forall u \in \L(E), ~ P(u) \in \L(E)$ car $(\L(E),+,\circ,\cdot)$ est une $\K$-algû´bre.
\end{defi}

\vspace*{-0.4cm}

\begin{prop}{Rû´gles de calcul.}{}
    Soit $E$ un $\K$-ev, $u\in\L(E)$, $P,Q\in\K[X]$, $\l\in\K$.
    \begin{enumerate}
        \item $(P+\l Q)(u) = P(u) + \l Q(u)$.
        \item $(P\times Q)(u)=P(u)\circ Q(u)$.
    \end{enumerate}
\end{prop}

\vspace*{-0.4cm}

\begin{nota}{}{}
    L'ensemble des polynûÇmes en $u$ : $\K[u]=\{P(u), ~ P\in\K[X]\}$.
\end{nota}

\vspace*{-0.4cm}

\begin{defi}{}{}
    Soit $P\in\K[X]$. On dit que c'est un \bf{polynûÇme annulateur} de $u$ ssi $P(u)=0_{\L(E)}$.
\end{defi}

\vspace*{-0.4cm}

\begin{thm}{}{}
    Soit $E$ un $\K$-ev et $u\in\L(E)$. On pose $\phi:P\mapsto P(u)$.
    \begin{enumerate}
        \item C'est un morphisme d'algû´bre.
        \item $\Im(\phi)=\K[u]$ est une algû´bre commutative.
        \item $\Ker(\phi)$ est un idûˋal de $\K[X]$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Trivial.\\
    \boxed{2.} $\Im(\phi)=\K[u]$ est une algû´bre comme image d'algû´bre par un morphisme d'algû´bre.\\
    \boxed{3.} $\Ker(\phi)$ est un idûˋal comme noyau d'un morphisme d'algû´bre.
\end{thm}

\vspace*{-0.4cm}

\begin{rappel}{}{}
    Tout idûˋal non-nul de $\K[X]$ est engendrûˋ par un unique polynûÇme unitaire.
\end{rappel}

\vspace*{-0.4cm}

\begin{defi}{}{}
    Soit $E$ un $\K$-ev, $I$ l'idûˋal des polynûÇmes annulateurs de $u$, si $I\neq(0)$, alors $I$ est engendrûˋ par un unique polynûÇme unitaire notûˋ $\Pi_u$ appelûˋ polynûÇme minimal de $u$.
\end{defi}

\vspace*{-0.4cm}

\begin{ex}{}{}
    Soit $E$ un $\K$-ev et $u$ une homothûˋtie non-nulle de $E$, alors $X-\l$ annule $u$.\\
    Donc $\Pi_u$ existe, aucun polynûÇme constant non nul n'annule $u$, donc $\Pi_u = X - \l$.
\end{ex}

\vspace*{-0.4cm}

\begin{ex}{}{}
    Soit $u$ projecteur de $E$ non-trivial, alors $u^2=u$ et $P=X^2-X$ est annulateur de $u$.
\end{ex}

\vspace*{-0.4cm}

\begin{prop}{$\star$}{}
    Soit $E$ un $\K$-ev et $u\in \L(E)$ tels que $\Pi_u$ bien dûˋfini.\\
    Alors $\deg\Pi_u \geq 1 \et \left[ \deg \Pi_u = 1 \iff u \nt{ homothûˋtie} \right]$.
    \tcblower
    Supposons $\Pi_n=\l\in\K$ par l'absurde, alors $\l\neq0$ car $\Pi_u\neq0_{\K[X]}$ puis $\Pi_u(u)=\l \Id_E \neq 0$, donc $\deg(\Pi_u)\geq1$.\\
    On a $u$ homothûˋtie $\ra$ $\exists \l \in \K, ~ \Pi_n = X - \l$ donc $\Pi_u(u)=0 \ra u-\l\Id_E = 0 \ra u = \l\Id_E$ homothûˋtie.
\end{prop}

\begin{prop}{Structure de $\K[u]$. $\star$}{}
    Soit $E$ un $\K$-ev et $u\in\L(E)$.
    \begin{enumerate}
        \item Si $u$ possû´de un polynûÇme annulateur non-nul, on note $d=\deg \Pi_u\in\N^*$.\\
        Alors $\K[u]$ est une $\K$-algû´bre de dimension $d$, et $(\Id_E, u, ..., u^{d-1})$ en est une base.
        \item Sinon, $\K[u]$ est une $\K$-algû´bre de dimension infinie, et $(u^n)_{n\in\N}$ est famille libre de $\K[u]$.
    \end{enumerate}
    \tcblower
    \boxed{1.} On sait dûˋjû  que $(\K[u],+,\circ,\cdot)$ est une $\K$-algû´bre.\\
    On pose $\B=(\Id_E,u,...,u^{d-1})$, c'est une famille de vecteurs de $\K[u]$.\\
    \bf{Libre.} Soient $(\l_0,..,\l_{d-1})\in\K^d, ~ \l_0\Id_E+...+\l_{d-1}u^{d-1}=0_{L(E)}$. C'est un polynûÇme de degrûˋ strictement infûˋrieur û  $d$ et annulant $u$, il est donc nul, donc $\forall k \in \lb0,d-1\rb, ~ \l_k=0$.\\
    \bf{Gûˋnûˋratrice.} Soit $v\in\K[u]$, $\exists P \in \K[X], ~ v=P(u)$, puis $P=\Pi_uQ + R$, ~ $Q,R\in\K[X]$, $\deg(R)<d$.\\
    Puis $P(u)=\cancel{\Pi_u(u)}\circ Q(u) + R(u) = R(u) \in \Vect(\Id_E,u,...,u^{d-1})$ car $\deg(R)<d$.\\
    C'est donc une base de $\K[u]$, qui est alors de dimension $d$.\\
    \boxed{2.} Soit $n\in\N$ et $(\l_0,...,\l_n)\in\K^{n+1}, ~ \l_0\Id_E+...+\l_n u^n = 0_{\L(E)}$.\\
    Alors $\l_0 + \l_1 X + ... + \l_n X^n$ annule $u$, il est donc nul, donc $\forall k \in \lb0,n\rb, ~ \l_k=0$.
\end{prop}

\begin{prop}{$\star$}{}
    Soit $E$ un $\K$-evdf, alors tout endomorphisme de $E$ admet un polynûÇme minimal.
    \tcblower
    Soit $u\in\L(E)$, et $n$ la dimension de $E$, alors $\dim(\L(E))=n^2$ donc toute famille libre de $\L(E)$ est de cardinal infûˋrieur û  $n^2$.
    En particulier, $(\Id_E, u, ..., u^{n^2})$ est une famille de $n^2+1$ vecteurs de $\L(E)$, elle est liûˋe.\\
    Alors $\exists (\l_0,...,\l_{n^2})\in\K$, tels que $\l_0\Id_E+...\l_{n^2}u^{n^2}=0_E$. On pose $P=\l_0 + ... + \l_{n^2}X^{n^2}$.\\
    On pose un polynûÇme minimal $\Pi_u$ et et $\deg \Pi_u \leq n^2$.
\end{prop}

\begin{defi}{PolynûÇme matriciel.}{}
    Soit $P=(a_nX^n)_n\in\K[X]$, et $A\in M_n(\K)$ pour $n\in\N^*$.
    \begin{equation*}
        P(A)=a_0 I_n + a_1 A + ... + a_NA^N \in M_n(\K)
    \end{equation*}
\end{defi}

\begin{prop}{}{}
    Toute matrice non-nulle $A$ admet un polynûÇme minimal notûˋ $\Pi_A$.
    \tcblower
    Propriûˋtûˋs :
    \begin{enumerate}
        \item $\Pi_A$ est le polynûÇme de plus bas degrûˋ qui annulle $A$.
        \item $P(A)=0_{M_n(\K)}\ra \Pi_A \mid P$.
        \item $\K[A]$ est une algû´bre commutative de $(M_n(\K),+,\times,\cdot)$, $\K[A]=\Vect(I_n,A,...,A^{d-1})$.
        \item $\deg \Pi_A = 1 \iff \exists \l \in \K, ~ A = \l I_n$.
    \end{enumerate}
\end{prop}

\begin{ex}{}{}
    Soient $A=\begin{pmatrix}1&2\\-1&1\end{pmatrix}, ~ B=\begin{pmatrix}0&1&-1\\2&0&4\\1&1&1\end{pmatrix}$. Dûˋterminer $\Pi_A$ et $\Pi_B$.
    \tcblower
    On a:
    \begin{equation*}
        A^2=\begin{pmatrix}
            -1 & 4 \\ -2 & -1
        \end{pmatrix}=2A - 3I_2.
    \end{equation*}
    Donc $\deg(\Pi_A)=2$, donc $\Pi_A=X^2-2X+3$.
    \begin{equation*}
        B^2 = \begin{pmatrix}
           1 & -1 & 3 \\ 4 & 6 & 2 \\ 3 & 2 & 4
        \end{pmatrix}
    \end{equation*}
\end{ex}\vspace*{-0.4cm}

\begin{prop}{Inverses.}{}
    \begin{enumerate}
        \item $u\in\L(E)$, admettant un polynûÇme minimal, alors $u\in \GL(E)\iff \Pi_u(0)\neq0$ puis $u^{-1}\in\K[u]$.
        \item $A\in M_n(\K)$ admettant un polynûÇme minimal, alors $A\in\GL_n(\K)\iff \Pi_A(0)\neq0 \et A^{-1}\in\K[A]$.
    \end{enumerate}
    \tcblower
    \boxed{1.\ra} Notons $\Pi_u=a_0+...+a_nX^n$, par l'absurde, sq $a_0=0$, alors $\Pi_u(u)=0_{\L(E)}\ra a_1u + ... + a_nu^n=0_{\L(E)}$.\\
    On compose û  gauche par $u^{-1}$ donc $a_1\Id+...+a_nu^{n-1}=0_{\L(E)}$. $P=a_1+...+a_nX^{n-1}$ est un polynûÇme annulateur de $u$, non nul, et $\deg(P)<n$, contredit la minimalitûˋ de $\Pi_u$.\\
    \boxed{1.\la} Notons $\Pi_u=a_0+...+a_nX^n$, $a_n\neq0$, alors $\Pi_u(u)=0_{\L(E)}\ra a_0\Id_E+...+a_nu^n=0_{\L(E)}$.\\
    Ainsi, $a_1u + ... + a_nu^n = -a_0\Id_{\L(E)}$ puis en posant $v=-\left( \frac{a_1}{a_0}\Id_E + ... + \frac{a_n}{a_0}u^{n-1} \right)$, $u\circ v= v \circ u = \Id_{\L(E)}$.\\
    On a montrûˋ que $u\in\GL(E)$ et $u^{-1}=v\in\K[u]$
\end{prop}

\begin{ex}{}{}
    Soit $A=\begin{pmatrix}1&0&2\\0&-1&3\\1&1&1\end{pmatrix}$, vûˋrifier que $A\in\GL_3(\R)$, calculer $A^{-1}$.
    \tcblower
    On a $\det(A)=-2$, donc $A\in\GL_3(\R)$.
    \begin{equation*}
        A^2 = \begin{pmatrix}
            3 & 2 & 4 \\ 3 & 4 & 0 \\ 2 & 0 & 6
        \end{pmatrix}; \quad A^3 = \begin{pmatrix}
            7 & 2 & 16 \\ 3 & -4 & 21 \\ 8 & 6 & 10
        \end{pmatrix}
    \end{equation*}
    On peut trouver que $A^3=A^2+6A-2I_3$, donc $(I,A,A^2,A^3)$ est une famille liûˋe.\\
    Donc $\Pi_A = 2 - 6X - X^2 + X^3$, et $\Pi_A(0)\neq0$, donc on retrouve $A\in\GL_3(\R)$.\\
    $A^{-1}=-\left( -3A - \frac{1}{2}A^2 + \frac{1}{2}I_3 \right)$. 
    \begin{equation*}
        A^{-1} = \begin{pmatrix} 2 & -1 & -1 \\ -3/2 & 1/2 & 3/2 \\ -1/2 & 1/2 & 1/2  \end{pmatrix}; \quad \nt{Com}(A) = \begin{pmatrix}-4&3&1\\2&-1&-1\\2&-3&-1\end{pmatrix} 
    \end{equation*}
\end{ex}

\begin{exercice}{}{}
    Soit $E$ un $\K$-evdf, $u\in\L(E)$, $P\in\K[X]$. Montrer que :
    \begin{equation*}
        P(u) \in \GL(E) \iff P\land \Pi_u = 1.
    \end{equation*}
    \bf{Rappel.} $\ov{k}\in\U(\Z/n\Z)$ ssi $k \land n = 1$.
    \tcblower
    \boxed{\la} Supposons $P\land \Pi_u = 1$ : $\exists U,V \in \K[X], ~ PU + \Pi_uV=1$ puis $P(u)\circ U(u)=\Id_E$. Donc $P(u)^{-1}=U(u)$.\\
    \boxed{\ra} Supposons $v=P(u)$ inversible, alors $v^{-1}\in\K[v]\subset\K[u]$ puis $\exists U \in \K[X], ~ P(u)^{-1}=v^{-1}=U(u)$.\\ Puis $P(u)\circ U(u)=\Id_E\ra(PQ-1)=0_{\L(E)}\ra \Pi_u \mid (PU-1) \ra \exists V \in \K[X], ~ PU-1=P_uV$.\\ Donc $\exists (U,V)\in\K[X]^2, ~ PU-\Pi_uV=1$ et par bûˋzout, $P\land \Pi_u = 1$.
\end{exercice}

\end{app}

\section{Sous-espaces stables.}

\begin{rappel}{}{}
    Soit $E$ un $\K$-ev, $F$ ssev de $E$, et $u:E\to E$. On dit que $F$ est stable par $u$ ssi $u(F)\subset F$.
\end{rappel}

\begin{prop}{}{puf}
    Soit $u\in\L(E)$, $F$ stable par $u$ et $P\in\K[X]$. Alors :
    \begin{equation*}
        \forall x \in F, ~ P(u)(x) = P(u_F)(x).
    \end{equation*}
    \tcblower
    Par rûˋcurrence sur $k\in\N$, $\forall x \in F, ~ u_F^k(x)=u^k(x)$.\\
    Puis $P=\sum_{k=0}^N a_kX^k$ et $P(u_F)(x)=\sum_{k=0}^N a_k u_F^k(x)=\sum_{k=0}^N a_ku^k(x)=P(u)(x)$.
\end{prop}

\begin{prop}{$\star$}{div}
    Soit $E$ un $\K$-ev, $u\in\L(E)$, $F$ stable par $u$.\\
    On suppose que $u$ possû´de un polynûÇme minimal $\Pi_u$.\\
    Alors l'endomorphisme induit $u_F$ aussi et $\Pi_{u_F} \mid \Pi_u$.
    \tcblower
    D'aprû´s la remarque prûˋcûˋdente, $\forall x \in F, ~ \Pi_u(u)(x)=\Pi_u(u_F)(x)$ donc $\Pi_u(u_F)=0_{\L(F)}$.\\
    Donc $u$ possû´de un polynûÇme annulateur $\Pi_u$ non-nul, donc $u_F$ possû´de un polynûÇme minimal et $\Pi_{u_F}\mid \Pi_u$.
\end{prop}

\begin{exercice}{}{}
    Soit $E$ un $\K$-ev, $F$ et $G$ deux ssev de $E$ tels que $E=F\oplus G$.\\
    Soit $u\in\L(E)$ tel que $F$ et $G$ soient stables par $u$. On suppose que $\Pi_u$ existe.\\
    Montrer que $\Pi_u=\PPCM(\Pi_{u_F},\Pi_{u_G})$.
    \tcblower
    Les deux endomorphismes induits $u_F$ et $u_G$ admettent des polynûÇmes minimaux, $\Pi_{u_F}\mid \Pi_u$ et $\Pi_{u_G} \mid \Pi_u$.
    Soit $P$ un multiple de $\Pi_{u_F}$ et de $\Pi_{u_G}$, $\exists (U,V) \in \K[X]^2, ~ P = U\Pi_{u_F} = V\Pi_{u_G}$ puis $P(u)=U(u)\circ \Pi_{u_F}(u)=V(u)\circ \Pi_{u_G}(u)$.\\
    Soit $x\in E$. $\exists!(y,z)\in F\times G, ~ x=y+z$ :
    \begin{align*}
        P(u)(x) &= P(u)(y) + P(u)(z) = U(u)\left( \Pi_{u_F}(u)(y) \right) + V(u)\left( \Pi_{u_G}(u)(z) \right)\\
        &= U(u)\left( \Pi_{u_F}(u_F)(y) \right) + V(u)\left( \Pi_{u_G}(u_G)(z) \right)\\
        &= U(u)(0_F) + V(u)(0_F) = 0_E.
    \end{align*}
    Donc finalement, $P(u)=0_{\L(E)}$, puis $\Pi_u\mid P$.
\end{exercice}

\begin{prop}{$\star$}{}
    Soit $E$ un $\K$-ev, et $u,v\in\L(E)$.
    \begin{enumerate}
        \item Si $u\circ v = v \circ u$, alors $\Ker u$ et $\Im u$ sont stables par $v$.
        \item Si $u\circ v=v\circ u$, alors $\forall P \in \K[X], ~ P(u)\circ v = v \circ P(u)$ donc $\Ker(P(u))$ et $\Im(P(u))$ sont stables par $v$.
        \item $\forall P \in \K[X], ~ \Im(P(u))$ et $\Ker(P(u))$ sont stables par $u$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Dûˋjû  vu.\\
    \boxed{2.} Supposons $u\circ v = v \circ u$, alors par rûˋcurrence, $\forall k \in \N, ~ u^k\circ v = v \circ u^k$.\\
    Soit $P=\sum_{k=0}^Na_kX^k\in\K[X]$, alors $P(u)\circ v =\sum_{k=0}^Na_k(u^k\circ v) = \sum_{k=0}^N a_k(v\circ u^k) = v\circ P(u)$.\\
    \boxed{3.} Appliquer \boxed{2.} avec $u=v$.
\end{prop}

\begin{ex}{}{}
    Soit $E$ un $\K$-ev. Soit $u\in\L(E)$ admettant un polynûÇme minimal $\Pi_u$.
    \begin{enumerate}
        \item Dans $\C$, montrer que $u$ admet une droite stable.
        \item Dans $\R$, montrer que $u$ admet une droite ou un plan stable.
    \end{enumerate}
    \tcblower
    \boxed{1.} On a $\deg(\Pi_u)\geq1$ et $\Pi_u\in\C[X]$ donc par d'Alembert-Gauss, $\exists Q \in \K[X], ~ \Pi_u = (X-\a)Q$.\\
    On a $\deg(Q)<\deg(\Pi_u)$ et $Q\neq0$ donc $Q$ n'annule pas $u$, i.e. $Q(u)\neq0_{\L(E)}~:~\exists x \in E, ~ y = Q(u)(x)\neq 0_E$.\\
    Puis $\Pi_u = (X-\a)Q \ra \Pi_u(u)=(u-\a\Id_E)\circ Q(u) \ra 0_E = (u-\a\Id)(Q(u)(x)) \ra u(y)=\a y$.\\
    Donc $\Vect(y)$ est une droite stable par $u$.\\
    \boxed{2.} De mûˆme, $\deg(\Pi_u)\geq1$ et $\Pi_u\in\R[X]$.\\
    $\hookrightarrow$ Si $\Pi_u$ admet une racine rûˋelle $\a\in\R$, on procû´de exactement comme dans le cas prûˋcûˋdent.\\
    $\hookrightarrow$ Sinon, $\deg(\Pi_u)\geq2$, il admet un diviseur $P$ de degrûˋ 2 : $\exists Q \in \R[X], ~ \Pi_u = PQ$.\\
    On $Q\neq0$ et $\deg(Q)<\deg(\Pi_u)$ donc $Q(u)\neq0_{\L(E)}$, $\exists x \in E, ~ y=Q(u)(x)\neq0$.\\
    Puis $\Pi_u=PQ\ra\Pi_u(u)=P(u)\circ Q(u)\ra (P(u)\circ Q(u))(x)=P(u)(Q(u)(x))=0_E$.\\ Or $P=aX^2+bX+c$ avec $a\neq0$. Donc $P(u)(y)=0$ avec $y\neq0_E$.\\
    Alors $P(u)(y)=0\iff au^2(y)+bu(y)+cy=0_E$ donc $\Vect(y, u(y))$ est stable par $u$.
\end{ex}

\begin{thm}{Dûˋcomposition des noyaux}{}
    Soit $E$ un $\K$-ev, $u\in\L(E)$, $P,Q\in\K[X]$ tels que $P\land Q=1$. Alors $\Ker~(PQ)(u)=\Ker~P(u) \oplus \Ker~Q(u)$.\\
    De plus, tous ces sous-espaces sont stables par $u$.
    \tcblower
    Par hypothû´se, $\exists (U,V)\in\K[X]^2, ~ PU+QV=1$, puis $P(u)\circ U(u)+Q(u)\circ V(u) = \Id_E$.\\
    Puis $(U(u)\circ P(u))(x) + (V(u)\circ Q(u))(x)=x$ i.e. $U(u)(\underbrace{P(u)(x)}_{0_E})+V(u)(\underbrace{Q(u)(x)}_{0_E})=x$, donc $x=0_E$.\\
    \boxed{\subset} Soit $x\in \Ker(PQ)(u)$. On note $y=U(u)(P(u)(x))$ et $z=V(u)(Q(u)(x))$ donc $x=y+z$.\\
    On a $Q(u)(y)=(U(u) \circ (PQ)(u))(x) = U(u)(0_E)=0_E$, donc $y\in\Ker(Q(u))$. (mûˆme raisonnement pour $z$)\\
    \boxed{\supset} Soit $x=y+z\in\Ker(P(u))+\Ker(Q(u))$.\\
    Alors $(PQ)(u)(x)=(PQ)(u)(y)+(PQ)(u)(z)=Q(u)(P(u)(y))+P(u)(Q(u)(z))=0_E$.
\end{thm}

\end{thm}

\begin{defi}{PolynûÇme caractûˋristique.}{}
    Soit $A\in M_n(\K)$. On pose $\forall \l \in \K, ~ P(\l)=\det(\l I_n-A)$.\\
    $P$ est un polynûÇme de degrûˋ $n$, unitaire, notûˋ $\X_A$ appelûˋ polynûÇme caractûˋristique de $A$.\\
    De plus, $\X_A = X^n-(\Tr(A))X^{n-1}+...+(-1)^n\det(A)$.
    \tcblower
    Soit $A=\begin{pmatrix}a&b\\c&d\end{pmatrix}\in M_2(\K)$.
    \begin{equation*}
        P(\l)=\det(\l I_2 - A) = \det\begin{pmatrix}\l - a & - b \\ -c & \l - d\end{pmatrix} = (\l-a)(\l-d)-bc=\l^2-(a+d)\l+ad-bc.\\
    \end{equation*}
    Soit $A=\begin{pmatrix}1&0&4\\2&-1&1\\1&1&2\end{pmatrix}\in M_3(\K)$.
    \begin{align*}
        P(\l)&=\det\begin{pmatrix}\l-1&0&-4\\-1&\l+1&-1\\-1&-1&\l-2\end{pmatrix} = (\l-1)\det\begin{pmatrix}\l+1&-1\\-1&\l-2\end{pmatrix}-4\det\begin{pmatrix}-2&\l+1\\-1&-1\end{pmatrix}\\
        &= (\l-1)(\l^2-\l-3) - 4(\l+3) = \l^3 - 2\l^2 - 6\l + 9 
    \end{align*}
    Cas gûˋnûˋral : on note $B=\l I_n-A$, $A=(a_{i,j})$ et $B=(b_{i,j})$.
    \begin{equation*}
        \X_A(\l) = \det(B) = \sum_{\s \in S_n}\e(\s)(\l\d_{\s(1),1}-a_{\s(1),1})...(\l\d_{\s(n),n}-a_{\s(n),n}).
    \end{equation*}
    Donc $\X_A$ est bien un polynûÇme en $\l$.
    \begin{align*}
        \X_A&=\sum_{\s\in S_n}\e(\s)(\d_{\s(1),1}X-a_{\s(1),1})...(\d_{\s(n),n}X-a_{\s(n),n})\\
        &=(X-a_{1,1})...(X-a_{n,n}) + \sum_{\s\neq\Id\in S_n}\e(\s)(\d_{\s(1),1}X-a_{\s(1),1})...(\d_{\s(n),n}X-a_{\s(n),n}).
    \end{align*}
    On remarque que si $\s\neq\Id$, alors $\exists i \in \lb1,n\rb$ tel que $\s(i)\neq i$, puis forcûˋment, $\exists j \in \lb1,n\rb, ~ j \neq i, ~ \s(j)\neq j$.\\
    Donc $\forall \s \in S_n, ~ \s\neq\Id, ~ \deg(Q_\s)\leq n-2$ puis $\deg\left( \sum_{\s\neq\Id\in S_n}\e(\s)Q_\s \right)\leq n-2$.\\
    Finalement :
    \begin{align*}
        \X_A &= (X-a_{1,1})...(X-a_{n,n})+R \quad \deg(R)\leq n-2\\
        &= X^n - (a_{1,1}+...+a_{n,n})X^{n-1} + T \quad \deg(T)\leq n-2\\
        &= X^n - \Tr(A)X^{n-1} + T \quad \deg(T) \leq n-2
    \end{align*}
    Il reste û  dûˋterminer le coefficient constant notûˋ $a_0$, or $a_0=\X_A(0)=\det(0I_n-A)=(-1)^n\det(A)$.
\end{defi}

\begin{prop}{$\star$}{}

\end{prop}

\begin{prop}{$\star$}{}

\end{prop}

\begin{thm}{Cayley-Hamilton. $\star$}{}
    \begin{enumerate}
        \item Soit $A\in M_n(\K)$, alors $\X_A(A)=0_{\M_n(\K)}$.
        \item Soit $E$ un $\K$-evdf et $u\in\L(E)$, alors $\X_u(u)=0_{\L(E)}$ et $\Pi_u(u)=0_{\L(E)}$.
    \end{enumerate}
    \vspace*{0.25cm}
    \begin{center}
        \fbox{Le polynûÇme caractûˋristique est un polynûÇme annulateur. Il est divisûˋ par le polynûÇme minimal.}
    \end{center}
    \tcblower
    (Non-exigible).\\
    On a $(1) \iff (2)$ et $\X_u(u) = 0_{\L(E)} \iff \Pi_u \mid \X_u$.\\
    Montrons que $\X_u(u) = 0_{\L(E)}$. Soit $x\in E$.\\
    $\hookrightarrow$ Vûˋrifions que $F=\Vect(u^k(x), ~ k\in\N)$ est stable par $u$.\\
    Soit $y \in F,~\exists N \in \N, ~ \exists (a_0, ..., a_N) \in \K^{N+1}, ~ y = a_0x + ... a_Nu_n(x)$ puis $u(y)=a_0u(x)+...+a_Nu^{N+1}(x)\in F$.\\
    $\hookrightarrow$ Cherchons une base de $F$, soit $p$ le plus petit entier tel que $(x, u(x), ..., u^p(x))$ soit liûˋe, existe car dim. finie.\\
    Par dûˋfinition de $p$, $\B=(x,u(x),...,u^{p-1}(x))$ est libre de $F$, vûˋrifions qu'elle est gûˋnûˋratrice.\\
    Dûˋjû , $u^p(x)\in\Vect(x,u(x),...,u^{p-1}(x))$ puis par rûˋcurrence, $\forall k \geq p, ~ u^k(x) \in \Vect(x,u(x),...,u^{p-1}(x))$.\\
    Finalement, $\forall k \in \N, ~ u^k(x) \in \Vect(x, u(x), ..., u^{p-1}(x))$ donc $F\subset\Vect(\B)$. Donc $\B$ base de $F$, donc $\dim F = p$.\\
    $\hookrightarrow$ Soit $x\in F$ donc $\forall k \in \lb0,p\rb, ~ u^k(x)=u^k_F(x)$ donc $\B=(x,u_F(x),...,u_F^{p-1}(x))$.\\
    Par dûˋfinition, $u_F$ est un endomorphisme cyclique, on a vu que $\X_{u_F}=\Pi_{u_F}$ donc $\X_{u_F}(u_F)=0_{\L(F)}$.\\
    D'autre part, $\X_{u_F} \mid \X_u$ donc $\exists Q \in \K[X], ~ \X_u = X_{u_F}Q$, puis $\X_u(u)=Q(u)\circ \X_{u_F}(u)$.\\
    Puis, $\X_u(u)(x)=_{(*)}Q(u)\left( \X_{u_F}(u)(x) \right)=Q(u) \circ \left(X_{u_F}(u_F)(x)\right) = Q(u)(0_E) = 0_E$.\\
    (*) Rappel : (\ref{prop:puf}).
\end{thm}

    Soit $E$ un $\K$-evdf de dimension $n$ et $u\in\L(E)$ tel que $\X_u$ est scindûˋ.\\
    On note $\X_u=(X-\l_1)^{\a_1}...(X-\l_p)^{\a_p}$ avec $\a_1+...+\a_p=n$ et $\a_i \geq 1$.\\
    On note $E_i=\Ker(u - \l_i\Id_E)$ sous-espaces propres et $F_i=\Ker(u-\l_i\Id_E)^n$ sous-espace caractûˋristiques associûˋs.\\
    On a $\forall i \in \lb1,p\rb, ~ E_i,F_i$ sont stables par $u$ et $E_i \subset F_i$ et $\dim F_i = \a_i$.
    \tcblower
    Dûˋjû , $E_i$ et $F_i$ sont stables par $u$ car $(u-\l_i\Id_E)^k$ commute avec $u$ pour tout $k\in\N$.\\
    Puis, $E_i \subset F_i$ (noyaux emboitûˋs) : $\forall f \in \L(E), \forall k \in \N, ~ \Ker f^k \subset \Ker f^{k+1}$.\\
    Il reste û  montrer que $\forall i \in \lb1,p\rb, ~ \dim F_i = \a_i$. $F_i$ est stable par $u$, on note $u_i$ l'endomorphisme induit.\\
    Alors $(u-\l_i \Id_E)^{\a_i}=P_i(u)\in\K[u]$ en posant $P_i=(X-\l_i)^{\a_i}$.\\
    Donc $\forall x \in F_i, ~ P_i(u_i)(x)=0_{F_i}, ~ P_i$ est aussi annulateur de $u_i$ : $\Sp(u_i)\subset\{$ racines de $P_i\}=\{\l_i\}$.\\
    Donc $\X_{u_i}=(X-\l_i)^{\dim F_i}$ ($\deg \X_{u_i}=\dim F_i$). Or $\X_{u_i}\mid \X_u$ donc $\forall i \in \lb1,p\rb, ~ \dim F_i \leq \a_i$.\\
    D'autre part, $\X_u$ est un polynûÇme annulateur de $u$ (Cayley-Hamilton).\\
    D'aprû´s dûˋcomposition des noyaux, $E=\Ker(\X_u(u))=\Ker(u-\l_1\Id)^{\a_1}\oplus...\oplus\Ker(u-\l_p\Id)^{\a_p}=F_1\oplus...\oplus F_p$.\\
    On passe aux dimensions : $\dim E = \sum_{i=1}^p\dim F_i$, d'autre part, $\dim E = \deg \X_u = \sum \a_i$\\
    Donc $\sum_{i=1}^p(\a_i-\dim F_i) = 0 \ra \forall i \in \lb1,p\rb,~ \dim F_i = \a_i$ car somme de positifs.
\end{prop}

\begin{prop}{}{}
    Soit $A\in M_n(\R)$. On peut considûˋrer que $A\in M_n(\C)$.\\
    $\bullet$ $\forall \l \in \K, ~ \X_A(\l)=\det(\l I_n - A)$.\\

\end{prop}

\begin{exercice}{}{}
    Soient $A,B,C\in\M_n(\K)$ telles que $AC=CB$. On note $r=\rg(C)$.\\
    Montrer que $\deg(\X_A\land\X_B)\geq r$.

    \begin{equation*}
        J_r = \begin{pmatrix}I_r & \Big| & 0_{r,n-r} \\ \hline 0_{n-r,r} & \Big| & 0_{n-r,n-r}\end{pmatrix}; \quad A' = \begin{pmatrix}A'_1 & \Big| & A'_2 \\ \hline A'_3 & \Big| & A'_4\end{pmatrix}; \quad B'=\begin{pmatrix} B'_1 & \Big| & B'_2 \\ \hline B'_3 & \Big| & B'_4\end{pmatrix}.
    \end{equation*}

        A'J_r = \begin{pmatrix}A'_1 & \Big| & (0) \\ \hline A'_3 & \Big| & (0)\end{pmatrix}; \quad J_rB' = \begin{pmatrix}B'_1 & \Big| & B'_3 \\ \hline (0) & \Big| & (0)\end{pmatrix}.

\end{exercice}
