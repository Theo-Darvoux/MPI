\section{Normes et distances.}

\begin{defi}{Norme. $\star$}{}
    Soit $E$ un $\K$-ev et $N:x\mapsto N(x)$ de $E$ dans $\R_+$. C'est une norme sur $E$ ssi :
    \begin{enumerate}
        \item $\forall x \in E, ~ N(x)=0 \iff x = 0_E$. (SÃĐparation.)
        \item $\forall x \in E, ~ \forall \lambda \in \K, ~ N(\lambda x) = |\lambda|N(x)$. (HomogÃĐnÃĐitÃĐ.)
        \item $\forall x,y \in E, ~ N(x+y) \leq N(x) + N(y)$. (InÃĐgalitÃĐ triangulaire.)
    \end{enumerate}
\end{defi}

\begin{ex}{}{}
    La valeur absolue est une norme sur $\R$, le module est une norme sur $\C$.
\end{ex}

\begin{ex}{}{}
    Montrer que toutes les normes sur $\R$ sont proportionnelles Ã  la valeur absolue.
    \tcblower
    Soit $\N$ une norme sur $\R$. On a $\forall x \in \R, \forall \l \in \R, ~ N(\l x)=|\l|N(x)$ donc $N(\l)=|\l|N(1)$.\\
    On a bien que $\forall \l \in \R, ~ N(\l)=a|\l|$.
\end{ex}

\begin{ex}{}{}
    Donner deux normes non proportionnelles sur $\C$.
    \tcblower
    Le module est une norme sur $\C$, $N(a+ib)=|a|+|b|$ aussi.
\end{ex}

\begin{prop}{GÃĐnÃĐralisation de l'inÃĐgalitÃĐ triangulaire.}{}
    Soit $E$ un $\K$-ev et $N$ une norme sur $E$, notÃĐe $\|\cdot\|$.
    \begin{enumerate}
        \item $\forall(x_1,...,x_n)\in E^n, ~ \|\sum_{i=1}^nx_i\|\leq\sum_{i=1}^n\|x_i\|$.
        \item $\forall x,y \in E, ~ |\|x\|-\|y\||\leq\|x-y\|$.
    \end{enumerate}
    \tcblower
    \boxed{1.} Par rÃĐcurrence facile sur $n$.\\
    \boxed{2.} Soient $x,y\in E$.
    \begin{equation*}
        \begin{cases}
            x=x-y+y\\
            y=y-x+x
        \end{cases} \ra \begin{cases}
            \|x\|\leq\|x-y\|+\|y\|\\
            \|y\|\leq\|y-x\|+\|x\|
        \end{cases} \ra \begin{cases}
            \|x\|-\|y\| \leq \|x-y\|\\
            \|y\|-\|x\| \leq \|x-y\|
        \end{cases}
    \end{equation*}
    Donc $|\|x\|-\|y\|| \leq \|x-y\|$.
\end{prop}

    \boxed{3.} $d(x,z)=\|x-z\|=\|x-y+y-z\|\leq \|x-y\| + \|y-z\| = d(x,y) + d(y,z)$.
\end{prop}

\section{Boules ouvertes, fermÃĐes, sphÃĻres.}

\begin{defi}{$\star$}{}
    Soit $(E,\|\cdot\|)$ un evn, $a\in E$ et $r\in\R_+$.
    \begin{enumerate}
        \item La boule ouverte de centre $a$ et de rayon $r$ est l'ensemble $B(a,r)=\{x\in E, ~ \|x-a\| < r\}$.
        \item La boule fermÃĐe de centre $a$ et de rayon $r$ est l'ensemble $B'(a,r)=\{x\in E, ~ \|x-a\|\leq r\}$.
        \item La sphÃĻre de centre $a$ et de rayon $r$ est l'ensemble $S(a,r)=\{x\in E, ~ \|x-a\|=r\}$.
    \end{enumerate}
    \bf{Remarque.} $B(a,r)\sqcup S(a,r) = B'(a,r)$ et $r_1<r_2 \ra B(a,r_1)\subset B(a,r_2)$.\\
    \bf{Remarque.} Dans $\R$, les boules sont des intervalles ouverts ou fermÃĐs.
\end{defi}

\begin{nota}{}{}
    On appelle $B(0,1)$ la boule unitÃĐ ouverte, $B'(0,1)$ la boule unitÃĐ fermÃĐe, et $S(0,1)$ la sphÃĻre unitÃĐ.
\end{nota}

\begin{defi}{Suite bornÃĐe. $\star$}{}
    Soit $(E,\|\cdot\|)$ un evn et $(u_n)_{n\in\N}$ une suite de $E$.
    \begin{center}

\begin{defi}{Convergence d'une suite. $\star$}{}
    Soit $(E,\|\cdot\|)$ un evn, $(u_n)_{n\in\N}$ une suite de $E$, et $l\in E$.
    \begin{center}

    \end{center}
\end{defi}

\end{ex}

\begin{prop}{UnicitÃĐ de la limite.}{}
    La limite d'une suite, si elle existe, est unique.
    \tcblower
    Supposons que $\lim u_n = l$ et $\lim u_n = l'$. Supposons par l'absurde que $l\neq l'$. On pose $\e=\frac{\|l-l'\|}{2}>0$.\\
    On a alors $\exists n_0 \in \N, ~ \forall n \geq n_0, ~ \|u_n - l\| < \e$ et $\exists n_1 \in \N, ~ \forall n \geq n_1, ~ \|u_n - l'\| < \e$.\\
    On pose $N=\max(n_0,n_1)$, donc $\forall n \geq N, ~ \|u_n-l\| < \e \et \|u_n - l\| < \e$.\\
    On a alors que $\|l-l'\| \leq \|u_n - l\| + \|u_n - l'\| < 2\e = \|l-l'\|$, absurde.
\end{prop}

\begin{thm}{CaractÃĻre bornÃĐ d'une suite convergente. $\star$}{}
    Toute suite convergente est bornÃĐe.
    \tcblower
    Soit $(u_n)$ une suite convergente de $(E,\|\cdot\|)$. On note $l=\lim u_n$.\\
    Avec $\e=1$, $\exists n_0\in\N, ~ \forall n \geq n_0, ~ \| u_n - l \| < 1$, donc $u_n \in B(l,1)$.\\
    On pose $r=\max \{1, \|u_1-l\|, ..., \|u_{n_0-1} - l\|\}$. On a $\forall n \geq n_0, ~ u_n \in B(l,r)$ car $\|u_n - l\| < 1 \leq r$.\\
    C'est vrai pour $0 \leq n \leq n_0 - 1$ car $\|u_n - l\| \leq r$.
\end{thm}
